# 03 【使用】

基本資料型態

# 1. String

APPEND key "value"： 追加字串

STRLEN key：取得字串長度

incr key：自增1

decr key：自減1

decr key 10：指定增加10

decr key 10：指定減少10

GETRANGE：擷取字串

set key "hello"
GETRANGE key 0 3 

// 擷取0到3的字串
// hell

set key "hello"
GETRANGE key 0 -1

// 擷取全部字串
// hello

SETRANGEx：替換

set key "abcd"
SETRANGE key 1 xx

// axxd

setex：設置過期時間

// 設定30秒後過期，字串為hello
setex key 30 "hello"

setnx：不存在則建立

set key redis
setnx key "mongodb"
// 0 (表示已存在，不會再設置)

mset：設置多個key

mset k1 v1 k2 v2 k3 v3

mget：取得多個key

mget k1 k2 k3

msetnx：不存在則建立，其中一個存在則不會建立

msetnx k1 v1 k4 v4

// 此時k1已經存在，k4不會設置

mset user:{id}:filed：

mset user:1:name tom user:1:age 18
mget user:1:name user:1:age

getset：如果值不存在，返回nil。存在，則設置值

getset db redis

# 2. List

# 1. LPUSH

# 將一個值或多個值，插入到列表頭部(左)

127.0.0.1:6379> LPUSH list one
(integer) 1
127.0.0.1:6379> LPUSH list two
(integer) 2
127.0.0.1:6379> LPUSH list three
(integer) 3

# 2. LRANGE

# 取得list中的值

127.0.0.1:6379> LRANGE list 0 -1
1) "three"
2) "two"
3) "one"
127.0.0.1:6379> LRANGE list 0 1
1) "three"
2) "two"
127.0.0.1:6379>

# 3. RPUSH

# 將一個值或多個值，插入到列表尾部(右)

127.0.0.1:6379> RPUSH list right
(integer) 4
127.0.0.1:6379> LRANGE list 0 -1
1) "three"
2) "two"
3) "one"
4) "right"

# 4. Lpop

# 1. 移除list的第一個元素

127.0.0.1:6379> Lpop list
"three"

# 2. 移除list的最後一個元素

127.0.0.1:6379> Rpop list
"right"

127.0.0.1:6379> LRANGE list 0 -1
1) "two"
2) "one"

# 5. lindex

# 取得list中的某一個值

127.0.0.1:6379> lindex list 1
"one"
127.0.0.1:6379> lindex list 0
"two"

# 6. Llen

# 返回list的長度

127.0.0.1:6379> LPUSH list one
(integer) 1
127.0.0.1:6379> LPUSH list two
(integer) 2
127.0.0.1:6379> LPUSH list three
(integer) 3

127.0.0.1:6379> Llen list
(integer) 3

# 7. Lrem

移除list集合中指定的value

# 1. 移除1個one

127.0.0.1:6379> LPUSH list one
(integer) 1
127.0.0.1:6379> LPUSH list two
(integer) 2
127.0.0.1:6379> LPUSH list three
(integer) 3
127.0.0.1:6379> Llen list
(integer) 3
127.0.0.1:6379> LPUSH list three
(integer) 4
127.0.0.1:6379> LRANGE list 0 -1
1) "three"
2) "three"
3) "two"
4) "one"

127.0.0.1:6379> Lrem list 1 one
(integer) 1
127.0.0.1:6379> LRANGE list 0 -1
1) "three"
2) "three"
3) "two"

# 2. 移除1個three

127.0.0.1:6379> Lrem list 1 three
(integer) 1
127.0.0.1:6379> LPUSH list three
(integer) 3

# 3. 移除2個three

127.0.0.1:6379> Lrem list 2 three
(integer) 2
127.0.0.1:6379> LRANGE list 0 -1
1) "two"

# 8. ltrim

# 擷取指定的長度

127.0.0.1:6379> Rpush list "hello"
(integer) 1
127.0.0.1:6379> Rpush list "hello1"
(integer) 2
127.0.0.1:6379> Rpush list "hello2"
(integer) 3
127.0.0.1:6379> Rpush list "hello3"
(integer) 4

127.0.0.1:6379> ltrim list 1 2
OK
127.0.0.1:6379> LRANGE list 0 -1
1) "hello1"
2) "hello2"

# 9. rpoplpush

# 移除列表的最後一個元素，移動到新的列表

127.0.0.1:6379> Rpush list "hello"
(integer) 1
127.0.0.1:6379> Rpush list "hello1"
(integer) 2
127.0.0.1:6379> Rpush list "hello2"
(integer) 3


127.0.0.1:6379> rpoplpush list otherlist
"hello2"
127.0.0.1:6379> LRANGE list 0 -1
1) "hello"
2) "hello1"
127.0.0.1:6379> LRANGE otherlist 0 -1
1) "hello2"

# 10. lset

將列表中指定下標的值替換為另外一個值

// 1. 判斷列表是否存在
127.0.0.1:6379> EXISTS list
(integer) 0

// 2. 不存在的話更新就會報錯
127.0.0.1:6379> lset list 0 item
(error) ERR no such key
127.0.0.1:6379> lpush list value1
(integer) 1
127.0.0.1:6379> LRANGE list 0 0
1) "value1"

// 3. 如果存在就會更新
127.0.0.1:6379> lset list 0 item
OK
127.0.0.1:6379> LRANGE list 0 0
1) "item"

// 4. 不存在的話更新就會報錯
127.0.0.1:6379> lset list 1 other
(error) ERR index out of range

# 11. LINSERT

將某個具體的value插入列表中的某個元素的前面或後面

127.0.0.1:6379> Rpush list "hello"
(integer) 1
127.0.0.1:6379> Rpush list "world"
(integer) 2

127.0.0.1:6379> LINSERT list before "world" "other"
(integer) 3
127.0.0.1:6379> LRANGE list 0 -1
1) "hello"
2) "other"
3) "world"
127.0.0.1:6379> LINSERT list after world new
(integer) 4
127.0.0.1:6379> LRANGE list 0 -1
1) "hello"
2) "other"
3) "world"
4) "new"

# 12. 總結

・ 實際上是一個鏈表，before Node after，left，right都可插入值

・ 如果key不存在，建立新的鏈表

・ 如果key存在，新增內容

・ 如果移除了所有值，空鏈表，也代表不存在

・ 在兩邊插入或改動值，效率最高。中間元素，相對效率較低

# 3. Set(集合)

Set中的值不能重複

# 1. sadd

set集合中增加元素

127.0.0.1:6379> sadd set "hello"
(integer) 1
127.0.0.1:6379> sadd set "world"
(integer) 1
127.0.0.1:6379> sadd set "!!!"
(integer) 1

# 2. SMEMBERS

查看set集合中的所有值

127.0.0.1:6379> SMEMBERS set
1) "!!!"
2) "world"
3) "hello"

# 3. SISMEMBER

判斷set集合中是否有某一個值

127.0.0.1:6379> SISMEMBER set hello
(integer) 1
127.0.0.1:6379> SISMEMBER set world
(integer) 1
127.0.0.1:6379> SISMEMBER set !!!
(integer) 1
127.0.0.1:6379> SISMEMBER set !!!!
(integer) 0

# 4. scard

取得set集合中元素的個數

127.0.0.1:6379> scard set
(integer) 3

# 5. srem

移除set集合中的指定元素

127.0.0.1:6379> srem set hello
(integer) 1
127.0.0.1:6379> scard set
(integer) 2
127.0.0.1:6379> SMEMBERS set
1) "!!!"
2) "world"

set集合：無序、值不重複

# 6. SRANDMEMBER

# 1. 隨機抽出set集合中的 1 個元素

127.0.0.1:6379> SMEMBERS list
1) "!!!"
2) "world"

127.0.0.1:6379> SRANDMEMBER list
"!!!"
127.0.0.1:6379> SRANDMEMBER list
"world"
127.0.0.1:6379> SRANDMEMBER list
"!!!"

# 2. 隨機抽出set集合中的 2 個元素

127.0.0.1:6379> SRANDMEMBER list 2
1) "!!!"
2) "world"

# 7. spop

隨機移除元素

127.0.0.1:6379> SMEMBERS set
1) "d"
2) "c"
3) "b"
4) "a"
127.0.0.1:6379> spop set
"d"
127.0.0.1:6379> spop set
"b"
127.0.0.1:6379> SMEMBERS set
1) "c"
2) "a"

# 8. smove

將一個指定的值，移動到指定的set集合中

127.0.0.1:6379> sadd set "hello"
(integer) 1
127.0.0.1:6379> sadd set "world"
(integer) 1
127.0.0.1:6379> sadd set "!!!!"
(integer) 1
127.0.0.1:6379> sadd set2 "set2"
(integer) 1
127.0.0.1:6379> smove set set2 hello
(integer) 1
127.0.0.1:6379> SMEMBERS set
1) "!!!!"
2) "world"
127.0.0.1:6379> SMEMBERS set2
1) "set2"
2) "hello"

# 9. SDIFF

查看集合中不同的值

(integer) 1
127.0.0.1:6379> sadd key1 b
(integer) 1
127.0.0.1:6379> sadd key1 c
(integer) 1
127.0.0.1:6379> sadd key2 c
(integer) 1
127.0.0.1:6379> sadd key2 d
(integer) 1
127.0.0.1:6379> sadd key2 e
(integer) 1

127.0.0.1:6379> SDIFF key1 key2
1) "a"
2) "b"

# 10. SINTER

查看集合中相同的值

127.0.0.1:6379> SINTER key1 key2
1) "c"

# 11. SUNION

查看所有集合中所有的值

127.0.0.1:6379> SUNION key1 key2
1) "c"
2) "a"
3) "b"
4) "d"
5) "e"

A用戶將所有關注的人放在一個set集合中，它的粉絲也放在一個集合中

共同關注、共同愛好

# 4. HASH

Map集合、key-Map集合

本質與String類型沒有太大區別，還是一個key-value

# 1. hset

set一個具體key-value

127.0.0.1:6379> hset myhash field1 hash1
(integer) 1

# 2. hget

127.0.0.1:6379> hget myhash field1 # 取得一個值
"hash1"

# 3. hmset

set多個具體key-value

127.0.0.1:6379>  myhash field hello field2 world #
OK

# 4. hmget

取得多個值

127.0.0.1:6379> hmget myhash field1 field2 # 
1) "hash1"
2) "world"

# 5. hgetall

取得所有值

127.0.0.1:6379> hgetall myhash
1) "field1"
2) "hash1"
3) "field"
4) "hello"
5) "field2"
6) "world"

# 6. hdel

刪除指定的值

127.0.0.1:6379> hdel myhash field1
(integer) 1
127.0.0.1:6379> hgetall myhash
1) "field"
2) "hello"
3) "field2"
4) "world"

# 7. hlen

取得hash表的字段數量

127.0.0.1:6379> hgetall myhash
1) "field"
2) "hello"
3) "field2"
4) "world"
127.0.0.1:6379> hlen myhash
(integer) 2

# 8. HEXISTS

判斷字段是否存在

127.0.0.1:6379> HEXISTS myhash field1
(integer) 0
127.0.0.1:6379> HEXISTS myhash field2
(integer) 1
127.0.0.1:6379> HEXISTS myhash field3
(integer) 0
127.0.0.1:6379> HEXISTS myhash field
(integer) 1

# 9. hkeys

取得所有key

127.0.0.1:6379> hkeys myhash
1) "field"
2) "field2"

# 10. hvals

取得所有value

127.0.0.1:6379> hvals myhash
1) "hello"
2) "world"

# 11. HINCRBY、DECRBY

指定增減量

127.0.0.1:6379> hset myhash field3 5
(integer) 1
127.0.0.1:6379> HINCRBY myhash field3 1
(integer) 6
127.0.0.1:6379> HINCRBY myhash field3 -1
(integer) 5

# 12. hsetnx

判斷是否存在，不存在則新增設置

127.0.0.1:6379> hsetnx myhash field4 hello
(integer) 1
127.0.0.1:6379> hsetnx myhash field4 hello
(integer) 0

# 13. 物件

Hash適合儲存經常變更的資料

Hash適合物件的儲存

String適合字串的儲存

127.0.0.1:6379> hset user:1 name redis
(integer) 1
127.0.0.1:6379> hget user:1 name
"redis"

# 5. Zset

在set的基礎上，增加了一個值

# 1. zadd

增加值

// 增加一個值
127.0.0.1:6379> zadd myset 1 one
(integer) 1

// 增加多個值
127.0.0.1:6379> zadd myset 2 two 3 three
(integer) 2
127.0.0.1:6379> ZRANGE myset 0 -1
1) "one"
2) "two"
3) "three"

# 2. ZRANGEBYSCORE

127.0.0.1:6379> zadd salary 2500 A
(integer) 1
127.0.0.1:6379> zadd salary 5000 B
(integer) 1
127.0.0.1:6379> zadd salary 1000 C
(integer) 1

// 從小到大排序
127.0.0.1:6379> ZRANGEBYSCORE salary -inf +inf
1) "C"
2) "A"
3) "B"

// 從大到小排序
127.0.0.1:6379> zrevrange salary 0 -1
1) "B"
2) "C"
127.0.0.1:6379> 

// 從小到大排序，附帶scores
127.0.0.1:6379> ZRANGEBYSCORE salary -inf +inf withscores
1) "C"
2) "1000"
3) "A"
4) "2500"
5) "B"
6) "5000"

// 小於2500的降序排列
127.0.0.1:6379> ZRANGEBYSCORE salary -inf 2500 withscores
1) "C"
2) "1000"
3) "A"
4) "2500"

# 3. zrem

移除元素

127.0.0.1:6379> zrange salary 0 -1
1) "C"
2) "A"
3) "B"
127.0.0.1:6379> zrem salary A
(integer) 1
127.0.0.1:6379> zrange salary 0 -1
1) "C"
2) "B"

# 4. zcard

取得集合中的個數

127.0.0.1:6379> zcard salary
(integer) 2

# 5. zcount

取得指定區間的成員數量

127.0.0.1:6379> zadd myset 1 hello
(integer) 1
127.0.0.1:6379> zadd myset 2 world 3 !!!!!
(integer) 2
127.0.0.1:6379> zcount myset 1 3
(integer) 3
127.0.0.1:6379> zcount myset 1 2
(integer) 2
