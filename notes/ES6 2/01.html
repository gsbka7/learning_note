# 01 【var、let、const】

# 1. 宣告

// var 可以重複宣告
// 可以修改值
var name = 'name1';
var name = 'name2';
console.log(name) // name2

// let 不可重複宣告
// 不可修改值
let name = 'name1';
let name = 'name2';
console.log(name) // Identifier 'name' has already been declared

// const 不可重複宣告
// 一般情況下不可修改值
const name = 'name1';
const name = 'name2';
console.log(name) // Identifier 'name' has already been declared

// const 值可變的情況
// 1. 陣列
const name = [];
name[0] = 'name1';
name[1] = 'name2';
console.log(name[0]) // name1
console.log(name[1]) // name2

// 2. 物件
const obj = {
    name = 'name1',
    age = 20;
}
obj.name = 'name2';
obj.age = 40;

let 沒有變數提升，但可以提前宣告

    var p1= document.querySelector('.p1');
    var p2= document.querySelector('.p2');

    p2.innerText = content; // 報錯，因為 let 沒有變數提升
    let content = 'p2'; 

    let content // 可以提前宣告
    p2.innerText = content; // 沒有值，會顯示 undefined
    content = 'p2';

const 沒有變數提升，不可提前宣告

    var p1= document.querySelector('.p1');
    var p2= document.querySelector('.p2');

    p2.innerText = content; // 報錯，因為 let 沒有變數提升
    const content = 'p2'; 

    const content // 不可以提前宣告
    p2.innerText = content; // 沒有值，會顯示 undefined
    content = 'p2';

# 3.作用域

# 1. 全域變數 與 區域變數

var name = 'nameOut'; // 全域變數
function func() {
    var name = 'nameInside'; // 區域變數
}
// 因為在全域下調用，所以會印出全域變數
console.log(name) // nameOut

# 2. if

ES5之前因為if和for都沒有區域作用域的概念，所以必須借助function的作用域來解決外部變數的問題

ES6中加入了let，使用let讓if和for產生作用域

var name = 'name1';
let age = 20;
if(true) { 
    var name = 'name2'; // if 沒有作用域的概念，即使 var 在{}內也不會形成作用域，所以代碼由上而下讀取會調用 if 內的 name
    console.log(name) // name2
    let age = 40; // 使用 let 產生{}內的作用域
    console.log(age) // 40
}
console.log(name) // name2
console.log(age) // 20

# 4.變數作用域

變數的可用範圍，限制某個變數只能在某個區域有效

使用 {} 來形成作用域，僅限於 let 和 const

   <p class="p1">p1</p>
   <p class="p2"></p>

使用作用域避免變數被修改

    var p1= document.querySelector('.p1');
    var p2= document.querySelector('.p2');

    // let 作用域
    {
        let content = 'p2';
        p2.innerText = content;
    }

    // const 作用域
    {
        const content = 'p2';
        p2.innerText = content;
    }

    // 因為 var 是在 let 和 const 的作用域外，所以值不會被修改
    var content = 'p1';

var 使用 {}，不會產生作用域，但可以使用 function 產生作用域

{
    var name = 'name1';
    console.log(name); // name1
}
    var name = 'name2'
    console.log(name); // name2

var age = 20;
function func() {
    var age = 40;
    return age;
}
console.log(func()); // 40，因為變數定義在函式的作用域中
console.log(age) // 20
