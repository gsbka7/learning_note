# 05 【 Express 】

# 1. 介紹

Express 是基於 Node.js 封裝的前後端網頁框架，特性為快速、開放、極簡。

Express 官方網站open in new window

# 2. 安裝

# 1. 建立 package.json

npm init

# 2. 安裝 express

$ npm install express --save

# 3. 路徑

const express = require('express') // 導入 express

const app = express() // 實體化

# 1. 基本使用

app.get('/', (req, res)=>{
    res.write('hello world')
    res.end()
})
app.listen(3000, ()=>{
    console.log('server start');
})

# 2. res.send 封裝方法

const express = require('express')

const app = express()

// req：前端傳來的資料，res：返回給前端的資料
app.get('/', (req, res)=>{
    res.write('hello world')
    res.end()
})
app.get('/login', (req, res)=>{
    res.send({ // send()是將 write()、end()進行封裝
        name: 'name',
        age: 20
    })

    // 不需設定 response header 用來解析內容
    res.send(`
        <html>
            <h1>login</h1>
        </html>
    `)

    res.send({
        name: 'name',
        age: 20
    })
})

app.listen(3000, ()=>{
    console.log('server start');
})

# 2.基本路徑

# 1. 可選 ?

// acd 和 abcd
app.get('/ab?cd', (req, res)=>{
    res.send('ab?cd')
})

# 2.佔位符 :

// ab固定
app.get('/ab/:id/:id2', (req, res)=>{
    res.send(':id/:id2')
})

# 3. 重複多次 +

// abcd abbcd abbbcd
app.get('/ab+cd', (req, res)=>{
    res.send('ab+cd')
})

# 4. 任意字符 *

// ab123cd abqwebc
app.get('/ab*cd', (req, res)=>{
    res.send('ab*cd')
})

# 5. 必選或不選 ()?

// abe abcde
app.get('/ab(cd)?e', (req, res)=>{
    res.send('ab(cd)?e')
})

# 6. 正則表達式

# 1. /
// 路徑中必須有a
app.get('/a/', (req, res)=>{
    res.send('/a/')
})
# 2. 結尾 $
// 結尾必須要是 fly
app.get('/*fly$', (req, res)=>{
    res.send('*fly$')
})

# 4. 中介軟體 Middleware

函數是一些有權存取要求物件 (req)、回應物件 (res) 和應用程式要求/回應循環中之下一個中介軟體函數的函數。下一個中介軟體函數通常以名為 next 的變數表示。

如果現行中介軟體函數不會結束要求/回應循環，它必須呼叫 next()，以便將控制權傳遞給下一個中介軟體函數。否則，要求將會停擺。

參考：Express-使用中介軟體open in new window

# 1. 基礎

# 1.基本使用

app.get('/home', (req, res, next)=>{ // 第一個函式：驗證token, cookie過期
    console.log('驗證token')
    const isValid = true
    if(isValid){
        next() // 如果驗證成功就直行進入下一步
    }else{
        res.send('error')
    }
},(req, res)=>{ // 第二個函式：查詢資料庫、返回內容
    res.send({list:[1, 2, 3]})
})

# 2. 方法

const func1 = (req, res, next)=>{
    console.log('驗證token');
    const isValid = true
    if(isValid){
        res.abc = 'fun1計算結果' // 設定新屬性
        next()
    }else{
        res.send('error')
    }
}
const func2 = (req, res)=>{
    console.log(res.abc); // 可以拿到新屬性的值
    res.send({list:[1, 2, 3]})
}

app.get('/home', [func1, func2]) // 經過 func1，驗證成功就執行next()進入func2
app.get('/list', [func1, (req, res)=>{
    res.send('list')
}])

# 2. 應用程式層次的中介軟體

應用程式層次的中介軟體綁定到 app物件 使用 app.use() 和 app.METHOD()

METHOD 是需要處理 HTTP請求 的方法，例如：GET、PUT、POST，全部使用小寫

// / 與 /login 擺放在app.use(func1)前面，不受限制
app.get('/', (req, res)=>{
    res.write('hello world')
    res.end()
})
app.get('/login', (req, res)=>{
    res.send({
        name: 'name',
        age: 20
    })
})

// 中介軟體函數
const func1 = (req, res, next)=>{
    console.log('驗證token');
    const isValid = true
    if(isValid){
        res.abc = 'fun1計算結果'
        next()
    }else{
        res.send('error')
    }
}

// 使用中介軟體 // 注意擺放的位置
app.use(func1)
app.use('/home', func1) // 只會響應 /home，其他不受限制，如：/list

// 擺放在app.use(func1)後面，要執行必須先經過 func1 驗證通過
------------------------------
const func2 = (req, res)=>{ 
    console.log(res.abc);
    res.send({list:[1, 2, 3]})
}

app.get('/home', [func2])
app.get('/list', (req, res)=>{
    res.send('list')
})
------------------------------

app.listen(3000, ()=>{
    console.log('server start');
})

LoginRouter.js

const express = require('express')

const router = express.Router()

// 路由器層次
// 響應前端的 get 請求
router.get('/', (req, res)=>{
    // 支持 send片段、json
    res.send("<h1>hello<h1>")

    // 支持 json
    res.json([1, 2, 3])

    // 渲染模板後返回給前端，只支援模板
    res.render('login', {error: '', isShow: false}) // 自動尋找 views 下的 login.ejs
    // error：於ejs模板中設定顯示
    // isShow：於ejs模板中設定變數
})

// 響應前端的 post 請求
router.post('/', (req, res)=>{
    if(req.body.username === 'username' && req.body.password === 'password') {
        console.log('登入成功');

        // 重定向到home
        res.redirect('/home')
    }else {
        console.log('登入失敗');
        res.render('login', {error: '用戶名密碼錯誤', isShow: true})
    }
})

module.exports = router
