server.js

const http = require('http')
const route = require('./route')

const api = require('./api')

const Router = {}

// Object.assign() 將兩個物件合併
function use(obj) {
    Object.assign(Router, obj)
}

function start() {
    http.createServer((req, res)=>{
        // 1. 啟動服務器
        const myURL = new URL(req.url, 'http://127.0.0.1')

        // 2. 依據前端的請求取得路徑執行方法渲染頁面
        try{
            // req：前端傳來的資料， res：後端響應的資料
            Router[myURL.pathname](req, res) // 取得 router.js 指定路徑的方法 
        }catch(error){ // 只要不是 router.js 裡的路徑，就會跳轉到404
            Router['/404'](req, res)
        }
    
    }).listen(80, ()=>{
        console.log('server start');
    })
}

// 將函式導出，由入口文件調用
exports.start = start
exports.use = use

router.js

const fs = require('fs')

function render(res, path, type="") {
    res.writeHead(200, {'Content-Type': `${type?type: 'text/html;charset=UTF-8'}`})
    res.write(fs.readFileSync(path), 'utf-8')
    res.end()
}

const route = {
    '/login': (req, res)=>{
        render(res, './static/login.html')
    },
    '/home': (req, res)=>{
        render(res, './static/home.html')
    },
    '/404': (req, res)=>{
        res.writeHead(404, {'Content-Type': 'text/html; charset=UTF-8'})
        res.write(fs.readFileSync('./static/404.html'), 'utf-8')
    },
    '/favicon.ico': (req, res)=>{
        render(res, './static/favicon.ico', 'image/x-icon')
    }
}
// 至 index.js 導入
module.exports = route

api.js

function render(res, data, type="") {
    res.writeHead(200, {'Content-Type': `${type?type: 'application/json'}`}) // json 格式
    res.write(data)
    res.end()
}

const apiRouter = {
    '/api/login':(req, res)=>{
        // 獲取參數方法
        const myURL = new URL(req.url, 'http://127.0.0.1')

        // get() 取得資料
        if(myURL.searchParams.get('username') === 'username' && myURL.searchParams.get('password') === 'password') {
            // 狀態：1：登入成功，0：登入失敗
            render(res, `{"ok":1}`) // json 格式
        }else{
            render(res, `{"ok":0}`)
        }
    },
    '/api/loginpost':(req, res)=>{
        // 獲取參數方法

        // 監聽資料
        var post = ''
        req.on('data', chunk=>{
            post+=chunk
        })

        // 資料蒐集結束
        req.on('end', ()=>{
            post = JSON.parse(post)
            if(post.username === 'username' && post.password === 'password') {
                render(res, `{"ok":1}`)
            }else{
                render(res, `{"ok":0}`)
            }
        })
    }
}

module.exports = apiRouter

const fs = require('fs')

// 拼接
const path = require('path')

// 自動判斷格式解析
const mime = require('mime')

function render(res, path, type="") {
    res.writeHead(200, {'Content-Type': `${type?type: 'text/html;charset=UTF-8'}`})
    res.write(fs.readFileSync(path), 'utf-8')
    res.end()
}
const route = {
    '/login': (req, res)=>{
        render(res, './static/login.html')
    },
    '/': (req, res)=>{ // 如果路徑為 127.0.0.1:3000/
        render(res, './static/home.html')
    },
    '/home': (req, res)=>{
        render(res, './static/home.html')
    },
    '/404': (req, res)=>{
        if(redaStaticFile(req, res)){
            return
        }else{
            res.writeHead(404, {'Content-Type': 'text/html; charset=UTF-8'})
            res.write(fs.readFileSync('./static/404.html'), 'utf-8')
        }
    },
    // 使用 mime.getType 處理路徑，從靜態資源路徑中讀取
    // '/favicon.ico': (req, res)=>{
    //     render(res, './static/favicon.ico', 'image/x-icon')
    // },
}

// 靜態資源管理
function redaStaticFile(req, res) {
    // 取得路徑
    const myURL = new URL(req.url, 'http://127.0.0.1')
    const pathname = path.join(__dirname, "/static" ,myURL.pathname); // 拼接出絕對路徑

    // if(myURL.pathname === '/') return false // 判斷路徑是否為127.0.0.1:3000/
    
    if(fs.existsSync(pathname)) { // fs.existsSync()：判斷路徑是否存在
        // 擷取出副檔名
        // console.log(myURL.pathname);                 // css/login.css
        // console.log(myURL.pathname.split('.'));      // [ '/css/login', 'css' ]
        // console.log(myURL.pathname.split('.')[1]);   // css

        render(res, pathname, mime.getType(myURL.pathname.split('.')[1]))
        return true
    }else{
        return false
    }
}
// 至 index.js 導入
module.exports = route
