# 09 【Provide、Inject】

如果從父組件向子組件傳遞資料，使用Props

但有一些嵌套更深的組件時，使用Props一層層往下傳，可能會很麻煩

這種情況，可以使用Provide、Inject，無論組件層次結構有多深，父組件都可以對其所有子組件提供數據

# 1. 基本使用

<template>
  <div>
    <Content/>
  </div>
</template>

<script>
import Content from '@/components/Content.vue'

export default {
  data() {
    return {
      message: "hello"
    }
  },
  // provide: {
  //   // 不能傳遞data裡的資料
  //   message: "hello"
  // },
  
  // 如果想要拿到data裡的值
  provide() {
    return{
      message: this.message
    }
  },
  components: {
    Content
  }
}
</script>

<template>
  <div>
    <div>
      <h2>Content</h2>
    </div>
  </div>
</template>

<script>
export default {
  name: 'Content',
  data() {
    return {}
  },
  inject: ['message']
}
</script>

# 2. Provide、Inject並不是響應式的

<template>
  <div>
    <div>
      <h2>Content</h2>
      <h2>子組件--{{message}}</h2>
    </div>
  </div>
</template>

<script>
export default {
  name: 'Content',
  data() {
    return {}
  },
  inject: ['message']
}
</script>

只有父組件會被改變

<template>
  <div>
    <Content/>

    <button @click="message='world'">改變message</button>
    <h2>父組件--{{message}}</h2>

  </div>
</template>

<script>
import Content from '@/components/Content.vue'

export default {
  data() {
    return {
      message: "hello"
    }
  },
  // provide: {
  //   // 不能傳遞data裡的資料
  //   message: "hello"
  // },
  
  // 如果想要拿到data裡的值
  provide() {
    return{
      message: this.message
    }
  },
  components: {
    Content
  }
}
</script>

# 3. 響應式方式

# 第一種

<template>
  <div>
    <Content/>

    <button @click="obj.message='world'">改變message</button>
    <h2>父組件--{{obj.message}}</h2>

  </div>
</template>

<script>
import Content from '@/components/Content.vue'

export default {
  data() {
    return {
      message: "hello",
      obj:{
        message: "hello"
      }
    }
  },
  provide() {
    return{
      // message: this.message
      obj: this.obj // 響應式
    }
  },
  components: {
    Content
  }
}
</script>

<template>
  <div>
    <div>
      <h2>Content</h2>
      <h2>子組件--{{obj.message}}</h2>
    </div>
  </div>
</template>

<script>
export default {
  name: 'Content',
  data() {
    return {}
  },
  // inject: ['message']
  inject: ['obj']
}
</script>

# ?第二種

<template>
  <div>
    <Content/>

    <button @click="message='message'">改變message</button>
    <h2>父組件--{{obj.message}}</h2>

  </div>
</template>

<script>
import Content from '@/components/Content.vue'

export default {
  data() {
    return {
      message: "hello",
      obj:{
        message: "world"
      }
    }
  },
  provide() {
    return{
      message:()=>this.message
    }
  },
  components: {
    Content
  }
}
</script>

<template>
  <div>
    <div>
      <h2>Content</h2>
      <h2>子組件--{{message()}}</h2>
    </div>
  </div>
</template>

<script>
export default {
  name: 'Content',
  data() {
    return {}
  },
  computed: {
  newMsg: function() {
    return this.message
    }
  },
  inject: ['message']
}
</script>
