# 1. 基本查找

# 範例1

查詢陣列中的某個元素是否存在

public class search {
    public static void main(String[] args) {
        int[] arr = {131, 127, 147, 81, 23, 7, 79};
        int num = 81;
        System.out.println(basicSearch(arr, 81));
    }

    public static boolean basicSearch(int[] arr, int num) {
        for (int i = 0; i < arr.length; i++) {
            if(arr[i] == num) {
                return true;
            }
        }
        return false;
    }
}

# 範例2

查詢某個元素的索引

import java.util.ArrayList;

public class search2 {
    public static void main(String[] args) {
        int[] arr = {131, 127, 147, 81, 23, 7, 79, 81};
        int num = 81;
        System.out.println(basicSearch(arr, 81));
    }

    // 如果要返回多個資料，可以把資料放到陣列或集合中
    public static ArrayList<Integer> basicSearch(int[] arr, int num) {
        ArrayList<Integer> list = new ArrayList<>();

        for (int i = 0; i < arr.length; i++) {
            if(arr[i] == num) {
                list.add(i);
            }
        }
        return list;
    }
}

# 2. 二分查找

# 範例

查詢某個元素在陣列中的索引'

public class search3 {
    public static void main(String[] args) {
        int[] arr = {131, 127, 147, 81, 23, 7, 79};
        System.out.println(binarySearch(arr, 81));

    }

    public static int binarySearch(int[] arr, int num) {
        // 1.定義兩個變數紀錄要查找的範圍
        int min = 0;
        int max = arr.length - 1;

        // 2.利用循環不斷去找要查找的資料
        while(true) {
            if(min > max) {
                return -1;
            }

            // 3.找到min和max的中間位置
            int mid = (min+max) / 2;

            // 4.用mid指向的元素跟要查找的元素比較
            if(arr[mid] > num) {
                // num 在 mid 的左邊
                // min不變，max = mid - 1
                max = mid - 1;
            }else if(arr[mid] < num) {
                // num 在 mid 的右邊
                // max不變，min = mid - 1
                min = mid + 1;
            }else{
                // num 跟 mid 指向的元素一樣
                // 表示找到
                return mid;
            }
        }
    }
}

# 總結

  1. 優勢：

提升查詢的效率

  1. 前提條件：

資料必須是有序的

  1. 過程

min和max表示目前要查詢的範圍

mid是在mix和max中間的

如果要查詢的元素在mid的左邊，縮小範圍時，mix不變，max等於mid減1

如果要查詢的元素在mid的右邊，縮小範圍時，max不變，min等於mid加1

# 3. 分塊查找

package demo1;

public class search4 {
    public static void main(String[] args) {
        // 1. 要把資料進行分塊
        // 長度18開根號 4.24
        int[] arr = {16, 5, 9, 12, 21, 18,
                32, 26, 45, 34, 50,
                48, 61, 52, 73, 66};

        // 2. 建立三個分塊的物件
        Blocka b1 = new Blocka(21, 0, 5);
        Blocka b2 = new Blocka(45, 6, 11);
        Blocka b3 = new Blocka(73, 12, 17);

        // 定義陣列用來管理三個分塊的物件(索引表)
        Blocka[] blockArr = {b1, b2, b3};

        // 定義一個變數用來記錄要查詢的元素
        int num = 32;
        
        // 調用方法
        int index = getIndex(blockArr, arr, num);

        System.out.println(index);
    }

    // 利用分塊查找的原理，查詢num的索引

    // 確定num在哪一塊中
    public static int getIndex(Blocka[] blockArr, int[] arr, int num) {
        // 1. 確定num在哪一塊中
        int indexBlock = findIndexBlock(blockArr, num);

        if(indexBlock == -1){
            // 表示num不在陣列之中
            return -1;
        }

        // 2.取得這一塊的起始索引和結束索引
        int startIndex = blockArr[indexBlock].getStartIndex();
        int endIndex = blockArr[indexBlock].getEndIndex();

        // 3. 遍歷
        for (int i = startIndex; i <= endIndex ; i++) {
            if(arr[i] == num) {
                return i;
            }
        }
        return -1;
    }


    // 定義一個方法，確定num在哪一塊中
    public static int findIndexBlock(Blocka[] blockArr, int num) {
        // 從0索引開始遍歷blockArr，如果num小於max，表示num在這一塊當中
        for (int i = 0; i < blockArr.length; i++) {
            if(num <= blockArr[i].getMax()) {
                return i;
            }
        }
        return -1;
    }


}

class Blocka{
    private int max; // 最大值
    private int startIndex; // // 開始索引
    private int endIndex; // 結束索引

    public Blocka() {}

    public Blocka(int max, int startIndex, int endIndex) {
        this.max = max;
        this.startIndex = startIndex;
        this.endIndex = endIndex;
    }

    public int getMax() {
        return max;
    }

    public void setMax(int max) {
        this.max = max;
    }

    public int getStartIndex() {
        return startIndex;
    }

    public void setStartIndex(int startIndex) {
        this.startIndex = startIndex;
    }

    public int getEndIndex() {
        return endIndex;
    }

    public void setEndIndex(int endIndex) {
        this.endIndex = endIndex;
    }
}

# 8. Lambda表達式

JDK8開始的一種新語法形式

# 規則

  1. 參數類型可以省略不寫

  2. 如果只有一個參數，參數類型可以省略，()也可以省略

  3. 如果方法體只有一行，大括號、分號、return可以省略不寫，但是需要同時省略

import java.util.Arrays;
import java.util.Comparator;

public class lambda1 {
    public static void main(String[] args) {
        Integer[] arr = {2, 3, 1, 5, 6, 7, 8, 4, 9};

        Arrays.sort(arr, new Comparator<Integer>() {
            @Override
            public int compare(Integer o1, Integer o2) {
                return o1 - o2;
            }
        });

        // lambda完整格式
        Arrays.sort(arr, (Integer o1, Integer o2)->{
                    return o1 - o2;
            }
        );

        // lambda省略寫法
        Arrays.sort(arr, (o1, o2)-> o1 - o2);

        System.out.println(Arrays.toString(arr));
    }
}

# 範例

匿名內部類調用方法 改寫成 lambda表達式

package demo1;

public class lambda2 {
    public static void main(String[] args) {

        // 1. 利用匿名內部類的形式去調用下面的方法
        // 調用一個方法的時候，如果形參是一個接口，要傳遞這個接口的實現類物件
        // 如果實現內部類只要用到一次，就可以用匿名內部類

        method(new Swim() {
            @Override
            public void swimming() {
                System.out.println("游泳");
            }
        });

        // 2. lambda表達式改寫
        method(
                // 函數式接口
                ()->{
                    System.out.println("正在游泳");
                }
        );
    }

    public static void method(Swim s) {
        s.swimming();
    }
}

// 只有接口可以進行lambda表達式改寫
interface Swim{
    public abstract void swimming();
}

# 總結

  1. 作用：

簡化函數式接口的匿名內部類的寫法

  1. 前提條件：

必須式接口的匿名內部類，接口中只能有一個抽象方法

  1. 好處：

Lambda是一個匿名函式，可以理解為是一段可以傳遞的程式碼，它可以寫出更簡潔、更靈活的程式碼，使java語言表達能力得到提升

不需要建立物件就寫方法體
