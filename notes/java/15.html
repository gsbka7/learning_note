# 15 【方法引用】

# 1. 介紹

把已經有的方法拿來使用，當作函數式接口中抽象方法的方法體

import java.util.Arrays;
import java.util.Comparator;

public class Demo {
    public static void main(String[] args) {
        // 將陣列按照倒敘進行排列
        Integer[] arr = {3,5,4,1,6,2};

        // 匿名內部類
        Arrays.sort(arr, new Comparator<Integer>() {
            @Override
            public int compare(Integer o1, Integer o2) {
                return o2 - o1;
            }
        });
        System.out.println(Arrays.toString(arr));

        // lambda表達式
        // Comparator是函數式接口
        Arrays.sort(arr, (Integer o1, Integer o2)->{
            return o2 - o1;
        });
        System.out.println(Arrays.toString(arr));

        // lambda表達式簡化
        Arrays.sort(arr, (o1, o2)->o2 - o1);
        System.out.println(Arrays.toString(arr));

        // 方法引用
        // 1. 引用處必須是函數是接口
        // 2. 被引用的方法需要已經存在
        // 3. 被引用的方法的形參和返回值需要和抽象方法的形參和返回值保持一致
        // 4. 被引用的方法的功能需要滿足目前的要求

        // 靜態方法要用類名調用
        // 表示引用Demo1這個類裡面的subtraction方法
        // 把這個方法當作抽象方法的方法體
        Arrays.sort(arr, Demo::subtraction);

        System.out.println(Arrays.toString(arr));
    }

    // 可以是Java已經寫好的，也可以是第三方的工具類
    public static int subtraction(int num1, int num2) {
        return num2 - num1;
    }
}

# 總結

  1. 方法引用是把已經有的方法拿來使用，當作函數式接口中抽象方法的方法體

  2. ::是方法引用福

  3. 方法引用注意點：

・ 引用處必須是函數是接口

・ 被引用的方法需要已經存在

・ 被引用的方法的形參和返回值需要和抽象方法的形參和返回值保持一致

・ 被引用的方法的功能需要滿足目前的要求

# 2. 分類

# 1. 引用靜態方法

格式：類名::靜態方法

Integer::parseInt

# 範例：

將String轉換成int

import java.util.ArrayList;
import java.util.Collections;
import java.util.function.Function;

public class Demo1 {
    public static void main(String[] args) {
        ArrayList<String> list = new ArrayList<>();
        Collections.addAll(list,"1", "2", "3", "4", "5");

        // String轉換成int
        list.stream().map(new Function<String, Integer>() {
            @Override
            public Integer apply(String s) {
                int i = Integer.parseInt(s);
                return i;
            }
        }).forEach(s-> System.out.println(s));

        // 函數式接口才可使用
        // 1. 方法需要已經存在
        // 2. 方法的形參和返回值需要跟抽象方法的形參和返回值保持一致
        // 3. 方法的功能需要把形參的字串轉換成整數
        list.stream().map(Integer::parseInt).forEach(s-> System.out.println(s));
    }
}

# 2. 引用成員方法

格式：物件::成員方法(形參、返回值完全一致)

  1. 其他類：其他類物件::方法名

  2. 本類：this::方法名

  3. 父類：super::方法名

this 與 super 的引用處不能是靜態方法

# 範例：

引用其他類的成員方法

過濾出A開頭，並且是3個字

import java.util.ArrayList;
import java.util.Collections;
import java.util.function.Predicate;

public class Demo1 {
    public static void main(String[] args) {
        ArrayList<String> list = new ArrayList<>();
        Collections.addAll(list, "A13", "A24", "B421", "D6555", "A7");

        // 過濾出A開頭，並且是3個字
        list.stream().filter(s->s.startsWith("A"))
                .filter(s->s.length() == 3)
                .forEach(s-> System.out.println(s));

        list.stream().filter(new Predicate<String>() {
            @Override
            public boolean test(String s) {
                return s.startsWith("A") && s.length() == 3;
            }
        }).forEach(s-> System.out.println(s));

        // 調用成員方法(使用其他類來調用)
        list.stream()
                .filter(new StringOperation()::stringJudge)
                .forEach(s-> System.out.println(s));

        // 靜態方法中沒有this，要建立本類的物件才能使用
        list.stream()
                .filter(new Demo1()::stringJudge)
                .forEach(s-> System.out.println(s));
    }

    public boolean stringJudge(String s) {
        return s.startsWith("A") && s.length() == 3;
    }
}

public class StringOperation {
    public boolean stringJudge(String s) {
        return s.startsWith("A") && s.length() == 3;
    }
}

# 3. 引用構造方法

格式：類名::new

Student::new

import java.util.ArrayList;
import java.util.Collections;
import java.util.List;
import java.util.function.Function;
import java.util.stream.Collectors;

public class Demo1 {
    public static void main(String[] args) {
        ArrayList<String> list = new ArrayList<>();
        Collections.addAll(list, "A,10", "B,25", "C,42", "D,65", "E,7");

        // 封裝成Student物件：String 轉換成 Student
        // 轉換好後再放到集合中
        List<Student> newList1 = list.stream().map(new Function<String, Student>() {
            @Override
            public Student apply(String s) {
                String[] arr = s.split(",");
                String name = arr[0];
                int age = Integer.parseInt(arr[1]);
                return new Student(name, age);
            }
        }).collect(Collectors.toList());
        System.out.println(newList1);

        // 引用構造方法
        List<Student> newList2 = list.stream()
                .map(Student::new)
                .collect(Collectors.toList());
        System.out.println(newList2);
    }
}

public class Student {
    private String name;
    private int age;

    public Student() {}

    // 需要和抽象方法的形參一致
    public Student(String str) {
        String[] arr = str.split(",");
        this.name = arr[0];
        this.age = Integer.parseInt(arr[1]);
    }

    public Student(String name, int age) {
        this.name = name;
        this.age = age;
    }

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public int getAge() {
        return age;
    }

    public void setAge(int age) {
        this.age = age;
    }

    @Override
    public String toString() {
        return "Student{" +
                "name='" + name + '\'' +
                ", age=" + age +
                '}';
    }
}

# 4. 類名引用成員方法

格式：類名::成員方法

String:substring

# 方法引用的規則(獨有)：

  1. 需要有函數式接口

  2. 被引用的方法必須已經存在

  3. 被引用方法的形參，需要跟抽象方法的第二個形參到最後一個形參保持一致

第一個參數：表示被引用方法的調用者，決定可以引用那些類中的方法

在Stream流當中，第一個參數一般都表示流裡面的每一個資料，假設流裡面的資料都是字串，使用這種方式進行方法使用，只能引用String這個類中的方法

第二個參數到最後到一個參數：跟被引用方法的形參保持一致，如果沒有第二個參數，說明被引用的方法是需要無參的成員方法

  1. 必引用方法的功能需要滿足目前的需求

  2. 侷限性：不能引用所有類中的成員方法

import java.util.ArrayList;
import java.util.Collections;
import java.util.function.Function;

public class Demo1 {
    public static void main(String[] args) {
        ArrayList<String> list = new ArrayList<>();
        Collections.addAll(list, "aaa", "bbb", "ccc", "ddd");

        // 變成大寫後輸出
        // String -> String
        list.stream().map(new Function<String, String>() {
            @Override
            public String apply(String s) {
                return s.toUpperCase();
            }
        }).forEach(s-> System.out.println(s));

        // 類名引用成員方法
        // 可以理解為：拿著流裡面的每一個資料，調用String類中的toUpperCase方法，方法的返回值就是轉換之後的結果
        list.stream().map(String::toUpperCase).forEach(s-> System.out.println(s));
    }
}

# 5. 引用陣列的構造方法

格式：資料類型[]::new

int[]::new

# 範例

集合中儲存一些整數，收集到陣列中

import java.util.ArrayList;
import java.util.Arrays;
import java.util.Collections;
import java.util.function.IntFunction;


public class Demo1 {
    public static void main(String[] args) {
        ArrayList<Integer> list = new ArrayList<>();
        Collections.addAll(list, 1, 2, 3, 4, 5);
        // 陣列的類型，必須跟流中資料的類型保持一致


        // 把資料收集到陣列中
        Integer[] arr1 = list.stream().toArray(new IntFunction<Integer[]>() {
            @Override
            public Integer[] apply(int value) {
                return new Integer[value];
            }
        });

        System.out.println(Arrays.toString(arr1));


        // 引用陣列的構造方法
        Integer[] arr2 = list.stream().toArray(Integer[]::new);
        System.out.println(arr2);
    }
}

# 3. 總結

  1. 方法引用是把已經存在的方法拿來使用，當作函數式接口中抽象方法的方法體

  2. ::是方法引用符

  3. 注意點

・ 需要有函數式接口

・ 被引用方法必須已經存在

・ 被引用方法的形參和返回值需要與抽象方法的保持一致

・ 被引用方法的功能需要滿足需求

# 3.1 分類

  1. 引用靜態方法

・ 類名::靜態方法

  1. 引用成員方法

・ 任意一個類：物件::成員方法

・ 本類：this::成員方法

・ 父類：super::成員方法

・ 本類和父類只能寫在非靜態方法中，因為靜態方法沒有this跟super

  1. 引用構造方法

・類名::new

  1. 使用類名引用成員方法

・ 類名::成員方法

・ 不能引用所有類中的成員方法

・ 如果抽象方法的第一個參數是A類型的，只能引用A類中的方法

  1. 引用陣列的構造方法

・ 資料類型[]::new

# 3.2 技巧

  1. 有沒有一個方法符合目前的需求

  2. 如果有這樣的方法，此方法是否滿足引用的規則

靜態：類名::方法名

成員方法：類名::方法名(流與引用方法在同一個類)，物件：方法名(方法在其他類)，this::方法名(方法在本類)，super::方法名(方法在父類)

構造方法：類名::new

# 4. 範例

# 1. 方法引用

集合中儲存一些字串的資料，收集到Student類型的陣列當中(使用方法引用)

import java.util.ArrayList;
import java.util.Arrays;
import java.util.Collections;

public class Demo1 {
    public static void main(String[] args) {
        // 集合中儲存一些字串的資料，收集到Student類型的陣列當中(使用方法引用)
        ArrayList<String> list = new ArrayList<>();
        Collections.addAll(list, "A,15", "B,25", "C,34", "D,21", "E,54");

        // 將字串變成Student物件，然後再把Student物件收集起來
        Student[] arr = list.stream().map(Student::new).toArray(Student[]::new);
        System.out.println(Arrays.toString(arr));
    }
}

# 2. 方法引用

建立集合增加Student物件，屬性：name，age

只取得姓名並放到陣列當中(使用方法引用)

import java.util.ArrayList;
import java.util.Arrays;
import java.util.function.Function;


public class Demo1 {
    public static void main(String[] args) {
        // 建立集合增加Student物件，屬性：name，age
        // 只取得姓名並放到陣列當中(使用方法引用)
        ArrayList<Student> list = new ArrayList<>();
        list.add(new Student("A", 23));
        list.add(new Student("B", 21));
        list.add(new Student("C", 20));

        // 把Student物件轉換成字串
        String[] arr = list.stream().map(new Function<Student, String>() {
            @Override
            public String apply(Student student) { // 抽象方法沒有第二個參數，所以引用的成員方法只能是空參的
                // 取得姓名
                return student.getName();
            }
            // map方法執行完後轉換成String，所以是String
        }).toArray(String[]::new);

        System.out.println(Arrays.toString(arr));

        // 類名::方法名：被引用的方髮型餐翅和抽象方法第二個參數後面保持一致
        String[] arr2 = list.stream().map(Student::getName).toArray(String[]::new);
        System.out.println(Arrays.toString(arr2));
    }
}

public class Student {
    private String name;
    private int age;

    public Student() {}

    public Student(String name, int age) {
        this.name = name;
        this.age = age;
    }

    // 此成員方法被調用
    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public int getAge() {
        return age;
    }

    public void setAge(int age) {
        this.age = age;
    }

    @Override
    public String toString() {
        return "Student{" +
                "name='" + name + '\'' +
                ", age=" + age +
                '}';
    }
}
