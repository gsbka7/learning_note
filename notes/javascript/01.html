# 01 【 JavaScript技巧 】

# 1. == 與 === 的區別

6 == 6;     // true
6 === 6;    // true
6 == "6";   // true
6 === "6";  // false

true == 1;     // true
true == 0;     // true
true === 1;    // false
false === 0;   // false

"" == 0;            // true
"  "  == 0;         // true
null == undefined;  // true，null：空值，undefined：有值但未定義

// 兩個不等於任何的值，在相等時也不能轉換成其他的值
null == 0;          // false
undefined == '';    // false

"false" == false; // false

// NaN表示不是數字，並不是確切的一個值，所以NaN不等於自身
NaN == NaN;      // false
NaN == false;    // false
NaN === false;   // false

var a = {};  // a 和 b 儲存在不同的stack(靜態記憶體)
var b = {};
var c = a;

a == b;    // false
a === b;   // false
a == c;    // true
a === c;   // true

# 2. 對調兩個變數

# 1. 臨時變數法

var a = 3;
var b = 5;

b = a;
a = c;

# 2. 加減法

var a = 3;
var b = 5;
a = a + b; // a = 8
b = a - b; // b = 3
a = a - b; // a = 3

# 3. 陣列法

var a = 3;
var b = 5;
a = [a, b];
b = a[0];
a = a[1];

# 4. 物件法

var a = 3;
var b = 5;
a = {a : b, b : a};
b = a.b;
a = b.a;

# 5. 陣列運算法

var a = 3;
var b = 5;
a = [b, b = a][0];

# 6. 解構賦值法

var a = 3;
var b = 5;
[a, b] = [b, a];

# 3. typeof 與 instanceof 的區別

# 1. typeof

返回一個小寫寫的類型字串

只需要一個操作數

必須為基本資料型別、函式、物件

typeof undefined; // undefined
typeof true;      // boolean
typeof 123;       // number
typeof NaN;       // number
typeof "string"   // string

// function
hello = function() {};
typeof hello;
typeof Symbol;
typeof new function();

// object
typeof [];
typeof {};
typeof null;
typeof new Array();
typeof new Error();
typeof new Number();
typeof new String();
typeof new Date();
typeof new Object();
typeof new RegEXP();

# 2. instanceof

a = new Number(123);
a instanceof Number;  // true

b = new String("string");
b = instanceof String; // true

c = new Array(1, 2, 3);
c = instanceof Array;  // true

d = new Object();
d = instanceof Object;  // true

必須為引用類型

基本類型會出現結果false

"string" = instanceof String;  // false
123 instanceof Number;  // false
true instanceof Boolean;  // false

[1, 2] = instanceof Array;  // true

var student = {name: "name", sex: "male"};

student instanceof Object;  // true

new Number(123) instanceof Number;  // true
typeofinstanceof
作用檢測資料類型檢測關聯性
返回小寫字串布爾值
操作數基本資料型別、函式、物件左邊必須是引用類型，右邊必須是函式
操作數量一個兩個

# 4. typeof NaN的結果是什麼

NaN(not a number) 表示是否屬於number型別的一種狀態：是或否，不是確認的值

console.log(typeof NaN); // number

var a = "abc";
console.log(Number(a)); // NaN，表達式中存在不可轉換的變數，返回無效的結果，不是返回確切的值

// NaN不等於本身，不是確切的值，代表一個範圍
console.log(NaN === NaN); // false

# 5. null 與 undefined 的區別

# 1. 兩者都是代表空值

null表示不應該有值

undefined表示應該有值，但還沒有定義

null轉為數值為0，undefined轉為數值為NaN(不是一個數字)

console.log(Number(null));
console.log(Number(undefined));

# 2. 何時會有null

  1. 函式中不傳任何參數時，該函數返回null
function greetObject(who) {
    if (!who) {
        return null;
    }
    return { message: `Hello, ${who}!` };
}

greetObject('Eric'); // { message: 'Hello, Eric!' }
greetObject();       // null
  1. 作為物件原型鏈的終點
console.log(Object.getPrototypeOf(Object.prototype));

# 3. 何時會有undefined

  1. 變數被宣告了，但是沒有賦值，就等於undefined
var a;
console.log(a);
  1. 調用函式時，應該提供的參數沒有提供，該參數就等於undefined
function fun(a) {
    console.log(a);
}
fun()
  1. 物件沒有賦值的屬性，該屬性的值就等於undefined
var obj = {};
console.log(obj.name);
  1. 函式沒有返回值時，默認返回undefined
var f = fun()
console.log(f);

# 6. hasOwnProperty

檢查物件內是不是存在某個屬性

const user = {
    name: 'admin'
    }

var hasName = user.hasOwnProperty("name");
console.log(hasName); // true

var hasEmail = user.hasOwnProperty("email");
console.log(hasEmail); // false

# in 運算子

in也能達到相同的目的

const user = {
    name: 'admin'
    }

console.log("name" in user); // true
console.log("email" in user); // false

# hasOwnProperty 與 in 的差別

# hasOwnProperty

只檢查在該物件裡面明確定義的屬性，就算對hasOwnProperty這屬性本身或其他所有物件上都有的預設方法(如toString)做檢查，都會得到false

const user = {
name: 'admin'
}
console.log(user.hasOwnProperty("name")); // true
console.log(user.hasOwnProperty("hasOwnProperty")); // false
console.log(user.hasOwnProperty("toString")); // false
# in

可以檢查到物件上所有可以被取用的屬性

const user = {
name: 'admin'
}
console.log("name" in user); // true
console.log("hasOwnProperty"in user); // true
console.log("toString" in user); // true

如果只是想確定物件內的方法是否能被使用，可以使用in

如果想去定物件內是否存在開發者主動定義的屬性，可以使用hasOwnProperty

# undfined

在JavaScript裡面，存取一個不存在的物件屬性，就會得到undefined，就像是變數在宣告之後沒有賦值就去存取它

const user = {
    name: 'admin'
}
const hasName = user.name === undefined // false
console.log(hasName);

    name: 'admin'
}
