# 05 【條件判斷】

# 1. v-if 基本使用

<template>
  <div id="app" >
    <!-- false不會顯示 -->
    <h2 v-if="true">{{message}}</h2>
    <h2 v-if="false">{{message}}</h2> 

    <!-- 綁定data裡的變數，判斷是否顯示 -->
    <h2 v-if="is_show">{{message}}</h2>
  </div>
</template>

<script>
export default {
  data: {
    message: 'hello',
    is_show: 'true'
  }
}
</script>

# 2. v-if、v-else

<template>
  <div id="app" >
    <!-- 如果is_show存在(true)，則顯示hello -->
    <h2 v-if="is_show">{{message}}</h2>

    <!-- 反之，顯示!is_show -->
    <h1 v-else>!is_show</h1>
  </div>
</template>

<script>
export default {
  data: {
    message: 'hello',
    is_show: 'true'
  }
}
</script>

# 3. v-if、v-else、v-else-if

<template>
  <div id="app">
    <h2 v-if="score>=90">優秀</h2>
    <h2 v-else-if="score>=80">良好</h2>
    <h2 v-else-if="score>=60">及格</h2>
    <h2 v-else>不及格</h2>

    <!-- 當條件很多時不建議使用v-else -->
    <h1>{{result}}</h1>
  </div>
</template>

<script>
export default {
    data: {
        score: 99,
    },
    computed: {
        result() {
            let showMessage = '';
            if(this.score>=90) {
                showMessage = '優秀';
            }else if(this.score>=80) {
                showMessage = '良好';
            }else if(this.score>=60) {
                showMessage = '及格';
            }else{
                showMessage = '不及格';
            };
            return showMessage;
        }
    }
}
</script>

# 4. 用戶切換案例(含虛擬DOM解說)

<template>
  <div id="app">
    <!-- 按下切換後，已經輸入的內容不會清空 -->
    <!-- Vue在進行DOM渲染時，出於性能考量，會盡可能復用已經存在的元素，而不是建立新的元素。所以會與虛擬DOM進行對比，然後再渲染到真實DOM上 -->
    <!-- 解決方法：加上key，賦予不同的值 -->
    <span v-if="isUser">
        <label for="username">帳號</label>
        <input type="text" name="username" id="username" placeholder="帳號" key="username">
    </span>
    <span v-else>
        <label for="email">信箱</label>
        <input type="text" name="email" id="email" placeholder="信箱" key="email">
    </span>
    <button @click="isUser = !isUser">切換類型</button>
  </div>
</template>

<script>
export default {
    data: {
        isUser: true
    }
}
</script>

# 3. v-show

<template>
  <div id="app">
    <h2 v-if="isShow">{{message}}</h2> <!-- 當條件為false時，不存在於dom中，相當於被註釋掉 -->
    <h2 v-show="isShow">{{message}}</h2> <!-- 當條件為false時，相當於添加行內樣式display: none -->
  </div>
</template>

<script>
export default {
     data: {
        message: 'hello',
        isShow: true
    }
}
</script>
