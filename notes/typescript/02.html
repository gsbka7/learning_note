# 02 【資料型別】

# 1. Boolean 布林

true or false

let flag:boolean = true;

flag = false

# 2. Number 數字

let a:number = 123;

a = 12.3

// 錯誤寫法，已經指定為number
a = "string";
a = false;

# 3. String 字串

let str:string = "hello";

str = "string"

// 錯誤寫法，已經指定為string
str = 12;
str = false;

# 4. Array 陣列

# 4.1 第一種定義方法

let arr:number[] = [1,2,3]; // 只能放入數字
let arr:number[] = [1,2,3,"str"]; // 放入字串會報錯

let arr:string[] = ["1","2","3"];

# 4.2 第二種定義方法

let arr:Array<number> = [1,2,3]　// 只能放入數字
let arr:Array<number> = [1,2,3,"str"]; // 放入字串會報錯

let arr:Array<string> = ["1","2","3"];

# 5. Tuple 元祖

屬於陣列的一種，可以指定多種型別

let arr:[string,number,boolean] = ["ts", 1, true];

# 6. Enum 列舉

enum Flag {success=1, error=-1};

let f:Flag = Flag.success;

console.log(f) // 1
console.log(Flag.error) // -1

enum Color {red, blue, orange};

let c:Color=Color.blue
console.log(c) // 1

enum Color {red, blue=5, orange};

let c:Color=Color.blue
console.log(c) // 5
console.log(orange) // 6，以上一個值為基準

# 7. Any 任意

let num:any = 123;

num = "str"; // 可以指定任一種型別

# 8. null、undefined

let num:number;
console.log(num); // 報錯，沒有賦值，undefined

let num:undefined;
console.log(num); // 沒有報錯，undefined

指定為 number | undefined

let num:number | undefined;
num = 123;
console.log(num); // 123

let num:number | undefined;
console.log(num); // undefined

指定為 null

let num:null;
num = null;

一個元素可能是 number | null | undefined

let num:number | null | undefined;
num = 123;
console.log(num); // 123

# 9. void

一般用於定義方法的時候方法沒有返回值

ES5 定義方法

function run() {
  console.log("run");
};
run();

表示方法沒有返回任何類型

function run():void {
  console.log("run");
};
run()

在 TypeScript 中預設的返回值為 undefined

function run():void {
};
let r = run();
r = 'undefined' // 正確
r = 'undefinedabcd' // 報錯

// 錯誤寫法
function run():number { // 返回的不是 number
  console.log("run");
};
run()

// 正確寫法
function run():number { // 返回的是 number
  console.log(123);
};
run()

# 10. never

是所有型別 (包含 null 和 undefined ) 的 子型別，代表不會出現的值

var a:never
a = 123; // 報錯

function error():never {
  throw new Error('Error')
}

let e:string = 'e';
e = error();
