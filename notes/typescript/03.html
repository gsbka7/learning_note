# 03 【函式】

# 1. 函式定義

# 1.1 ES5

一般函式

function func() {
    return 'func';
};

匿名函式

var func2 = function() {
    return 'func2';
};

# 1.2 TypeScript

函式宣告

function func():string {
    return 'func';
};

匿名函式

var func2 = function():number {
    return 123;
};

傳遞參數

・一般函式

function getInfo1(name:string,age:number):string { // string 為返回的型別
    return `${name}-----${age}`
}

・逆名函式

var getInfo2=function(name:string,age:number):string {
    return `${name}-----${age}`
}

・沒有返回值

function fun3():void {
    console.log('fun3');
}

# 2. 可選參數

加上 ? 表示參數可傳可不傳

可選參數必須要放在所有參數的最後面

function getInfo3(name:string,age?:number):string {
    if(age){
        return `${name}-----${age}`
    }else{
        return `${name}-----age`
    }
}

# 3. 默認參數

默認參數如果有設定值，將以設定的值為準

function getInfo4(name:string,age:number=20):string {
    if(age){
        return `${name}-----${age}`
    }else{
        return `${name}-----age`
    }
}

# 4. 剩餘參數

原本的函式

function sum(a:number,b:number,c:number):number {
    return a + b + c;
}

sum(1, 2, 3);

使用三點運算符

function sum(...result:number[]):number {
    var sum = 0
    for(var i = 0;i < result.length;i++){
        sum += result[i];
    }
    return sum;
}
sum(1, 2, 3, 4, 5, 6);

function sum(a:number,b:number,...result:number[]):number {
    var sum = 0
    for(var i = 0;i<result.length;i++){
        sum += result[i]
    }
    return sum
}
sum(1, 2, 3, 4, 5, 6)

# 5. 函式重載

# 5.1 ES5

出現相同名稱的方法，下面會替換掉上面的

function css(config) {

};
function css(config,value) {

};

# 5.2 TS

function getInfo5(name:string):string;
function getInfo5(age:number):number;
function getInfo5(str:any):any {
    if(typeof str==='string'){
        return 'name：' + str
    }else{
        return 'age：' + str
    }
};
console.log(getInfo5('name')); // 執行第一個方法
console.log(getInfo5(20)); // 執行第二個方法
console.log(getInfo5(true)); // 錯誤
