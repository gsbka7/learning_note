# 02 【 基礎 】

# 1. 語法

使用 <?php ?> 包括程式碼，以 ; `結尾

<?php echo "Hello World"?>;

# 2. 輸出函式

・ echo：可以輸出多個參數

・ print：只能輸出一個參數

・ echo print：只能輸出一個參數，會有回傳值1

ehco的執行速度會比print快

<?php
    echo "Hello", "World"; // 可帶入兩個參數 //  Hello World
    echo "<BR>";
    print "Hello"; // 只能帶入一個參數 // Hello
    echo "<BR>";
    echo print "Hello"; // Hello1
?>

# 3. 資料型態，變數，常數

# 1. 常見資料型態

# 1. 整數(integer)

長度：-2147483648 ~ 2147483647

除了十進位表示外，八進位：數值前面加上0，數值十六進位：前面加上0x

# 2. 浮點數(floating point number)

長度：1.7E-308 ～ 1.7E+308

可為正負的浮點數

# 3. 字串(string)

使用''或""包括

使用''或""包括變數所顯示的結果不同

<?php
    $name = "Jason";
    $str1 = "Hello! $name" // Hello! Jason
    echo "<BR>";
    $str1 = 'Hello! $name' // Hello! $name
?>

# 4. 布林(boolean)

其值有true、flase兩種，判斷條件是否成立

# 5. 空值(null)

指變數內無任何資料

<?php
    $num = null;
    echo $num; // 不會印出任何結果
?>

# 6. 強制轉換

變數可以透過兩種方式進行轉換

# 1. 在變數或運算子錢加上指定資料型態強制進行轉換
轉換型態說明
(int) (integer)移除小數的部分，轉型為整數
(bool) (boolean)轉型為布林
(float) (double) (real)轉型為浮點數
(string)轉型為字串
(array)轉型為陣列
(object)轉型為物件

浮點數轉換為整數

<?php
    $num1 = 10.1;
    echo $num1;
    echo "<BR>";
    $num2 = (int)$num1;
    echo $num2;
?>
# 2. settype 函式

浮點數轉換為整數

<?php
    $num1 = 10.1;
    echo $num1;
    echo "<BR>";
    settype($num1, "int");
    echo $num1;
?>

# 2. 變數

變數可以暫存資料，使用時會在變數名稱前加上 $ ，被引用的變數還沒設定數值前內容為空值

PHP 是非強制檢查型別的程式語言，變數在引用時不需要事先宣告資料型態；資料型態會自動依照變數的內容決定

規則：

・ 字首必須是英文大小寫字母或是底線

・ 第二個字元後可以是(1)：英文大小寫，(2)：數字，(3)：底線

・ 長度沒有限制

・ 無法使用PHP保留字

# 1. 區域變數

在特定區域範圍中才能使用的變數，不能在其他函式中使用；函式的變數若是區域變數，也不能在其他區域使用

<?php
    $num1 = 10;
    function func() {
        $num2 = 20;

        echo $num1; // 出現錯誤
        echo "<BR>";
        echo $num2; // 20
    };

    func();

    echo $num1; // 10
    echo "<BR>";
    echo $num2; // 出現錯誤
?>

# 2. 全域變數

在任何函式範圍內皆可使用

PHP是在函式中利用全域指令來引用在該函式外的變數：

  1. 在該函式中將變數宣告為global
<?php
    $num = 10;
    function func() {
        global $num;
        echo $num; // 10
        echo "<BR>";
    }
    func();
    echo $num; // 10
?>
  1. 利用 GLOBAL 陣列
<?php
    $GLOBALS['a'] = 10;
    function func() {
        echo $GLOBALS['a']; // 10
        echo "<BR>";
    }
    func();
    echo $GLOBALS['a']; // 10
?>

# 3. 靜態變數

即是變數的值，不會因為離開函式而消失，靜態變數會保留最後使用或修改的值

使用保留字 static 宣告

function func() {
    static $num = 10;
    $num = $num + 1;
    echo $num;
    echo "<BR>";
}

/* 結果為
    1
    2
    3
*/

# 4. 動態變數

一種變數參照，透過某一變數名稱來當作另一個變數的參照

使用 $$ 宣告

<?php
    $A = "AAA";
    $Z = "ZZZ";
    $store = "A"; // 宣告 store 為 變數A 的名稱
    echo $$store; // AAA

    echo "<BR>";

    $store = "Z"; // 宣告 store 為 變數Z 的名稱
    echo $$store; // ZZZ
?>

# 5. 預設變數

主要是與伺服器或使用者環境相關的系統變數

常見之預設變數：

預設變數說明
$_SERVER儲存伺服器內或使用者環境的資訊陣列變數
$_POST儲存以 POST 方式傳入的陣列變數
$_GET儲存以 GET 方式傳入的陣列變數
$_COOKIE儲存以 COOKIE 方式註冊的陣列變數
$_SESSION儲存以 SESSION 方式註冊的陣列變數
$_FILES儲存以 POST 方式傳入檔案的陣列變數
$_REQUEST儲存以 POST、GET、COOKIE、FILES 方式傳入的陣列變數

# 3. 常數

定義一些無法改變的數值內容；分為內建常數和自定義數

內建常數是系統內以定義的常數工具，主要是提供給程式開發者使用通常名稱為大寫

# 1. 常見的內建常數：

內建常數說明
PHP_VERSIONPHP版本
PHP_OS執行 PHP 的系統
TRUE真值
FLASE假值
E_ERROR指到最近的錯誤，並中斷程式產生報告，值為 1
E_WARNING指到最近的警告，不會中斷程式，值為 2
E_PARSE分析語法錯誤之處，值為 4
E_NOTICE用於不尋常但不一定是錯誤之處，值為 8

# 2. 自定義數

語法：define(常數名稱, 常數值)

<?php
    define("PI", 3.1415); // 定義 PI 為3.1415
    echo $round_area = 3 * 3 * PI // 計算圓面積
?>

# 3. 魔術常數

內建常數說明
LINE目前這一行程是在檔案中的行數
FILE完整路徑與檔案名稱
DIR檔案目錄的完整路徑
FUNCTION函式名稱
CLASS類別名稱
METHOD類別的方法名稱
NAMESPACE目前命名空間的名稱

# 4. 運算子

用來計算以及判斷條件

# 1. 算數運算子

運算子說明
+加
-減
*乘
/除
%取餘數

# 2. 遞增運算子、遞減運算子

運算子說明
++遞增
--遞減
<?php
    $num = 10;
    echo $num++; // 11
    echo "<BR>";
    echo $num--; // 9
?>

# 3. 指定運算子、複合運算子

運算子說明範例運算拆解
=指定變數的數值$a = 10$a = 10
+=變數與數值相加$a += 10$a = $a + 10
-=變數與數值相減$a -= 10$a = $a - 10
*=變數與數值相乘$a *= 10$a = $a * 10
/=變數與數值相除$a /= 10$a = $a / 10
%=變數與數值相取餘數$a %= 10$a = $a % 10
.=變數與字串串接$a .= 70$a = $a.70
<<=左移 x 位元$a <<= 1$a = $a << 1
>>=右移 x 位元$a >>= 2$a = $a >> 1
&=相互位元 and 運算$a &= $b$a = $a & $b
|=相互位元 or 運算$a |= $b$a = $a | $b
^=相互位元 xor 運算$a ^= $b$a = $a ^ $b
<?php
    $num = 3;
    echo $num += 4; // 7
    echo "<BR>";
    echo $num <<= 2; // 28
    echo "<BR>";
    echo $num &= 3; // 0
    echo "<BR>";
    echo $num .= "~Hello"; // 0~Hello
?>

# 4. 比較運算子

運算子意義
==等於
===全等於
!=不等於
<>不等於
!==非等於
>大於
<小於
>=大於等於
<>=小於等於

# 5. 邏輯運算子

運算子意義
&&(and)條件 and 運算
!(not)條件 not 運算
xor條件 xor 運算

  1. && 只有兩邊結果皆為 true 時，運算結果才為 true

  2. || 只要其中一邊結果為 true，運算結果就是 true

  3. ! 為真假轉換

  4. xor 為當兩個結果相異時，運算結果才為 true

# 6. 位元運算子

運算子意義範例
&位元 AND 運算$a & $b
|位元 OR 運算$a | $b
^位元 XOR 運算$a ^ $b
~位元 NOT 運算~$a
<<位元左移$a << 2
>>位元右移$a >> 1
<?php
    $num1 = 6;
    $num2 = 3;

    echo $num1 & $num2; // 2
    echo "<BR>";
    echo $num1 | $num2; // 7
    echo "<BR>";
    echo $num1 ^ $num2; //5
    echo "<BR>";
    echo ~$num1; // -7
    echo "<BR>";
    echo $num1 << $num2; // 48
    echo "<BR>";
    echo $num1 >> $num2; // 0
?>

# 7. 字串連接運算子

使用 . 連接字串

<?php
    echo "Hello" . "World"; // HelloWorld
?>

# 4. 條件敘述

# 1. if

判斷條件是否成立，成立則執行程式碼

<?php
    if(判斷條件) {
        程式碼
};
?>

# 2. if...else

判斷條件是否成立，成立則執行程式碼1，不成立則執行程式碼2

<?php
    if(判斷條件) {
        程式碼1
    }else{
        程式碼2
    };
?>

# 3. if...elseif

判斷條件是否成立，成立則執行程式碼1，不成立則執行再向下尋找符合的條件，執行程式碼

<?php
    if(判斷條件) {
        程式碼1
    }elseif{
        程式碼2
    }else{
        程式碼3
    };
?>

# 4. switch

switch 常被用來簡化 elseif

case內的break為跳脫命令，用於在找到符合的結果時結束執行；若沒有加上 baeak ，即使找到符合的結果也會繼續向下執行

<?php
    switch(變數、常數或運算式) {
        case '結果1';
            程式碼1;
            break;
        case '結果2';
            程式碼2;
            break;
        default: // 預設條件，上面的條件皆不成立時執行
            程式碼3;
    }
?>

# 5. 迴圈

# 1. for

計算1累加到100的結果

<?php
    $sum = 0;
    for($i = 1; $i <= 100; $i++) {
        $sum += $i;
    };
    echo $sum
?>

# 2. while

判斷條件成立才執行迴圈

計算1累加到100的結果

<?php
    $sum = 0;
    $i = 1;
    while($i <= 100) {
        $sum += $i;
        $i++;
    }
    echo $sum;
?>

# 3. do while

與while差別於不管是否符合條件，都會先執行一次

<?php
    $i = 1;
    do{
        echo "迴圈執行第".$i."次"; // 雖然條件不符合，但會被執行一次
        $i++;
    }while($i>3);
?>

# 4. 巢狀迴圈

指一個迴圈內部有一個或多個迴圈

<?php
    $i = 1;
    while($i <= 2) {
        for($j = 1; $j <= 3; $j++) {
            echo "while迴圈第".$i."次，for迴圈第".$j."次<BR>";
        };
        $i++;
    };
?>

/*
    while迴圈第1次，for迴圈第1次
    while迴圈第1次，for迴圈第2次
    while迴圈第1次，for迴圈第3次
    while迴圈第2次，for迴圈第1次
    while迴圈第2次，for迴圈第2次
    while迴圈第2次，for迴圈第3次
*/

# 5. break 與 continue

break 是指中斷迴圈執行

<?php
    $i = 1;
    while($i <= 99) {
        echo "while迴圈執行第".$i."次<BR>";
        if($i == 3) {
            echo "while迴圈break跳出";
            break;
        }
        $i++;
    }
?>

/*
    while迴圈執行第1次
    while迴圈執行第2次
    while迴圈執行第3次
    while迴圈break跳出
*/

continue 可以讓迴圈略過

<?php
    for($i = 1; $i <= 10; $i++) {
        if($i < 7) {
            continue;
        }
        echo "while迴圈執行第".$i."次<BR>";
    }
?>

/*
    while迴圈執行第7次
    while迴圈執行第8次
    while迴圈執行第9次
    while迴圈執行第10次
*/
