# 09 【 範例 - 留言板 】

文章CRUD功能

dbc.php：各項功能函式

env.php：連線環境變數

index.php：文章列表首頁

form.html：新增文章頁面

create_blog.php：新增文章寫入資料庫

detail.php：文章詳細頁面

update_form.php：文章編輯頁面

update.php：編輯完成的文章寫入資料庫

delete.php：刪除文章

# dbc.php

<?php
    // 建立資料庫連線
    function connect() {
        $host = DB_HOST;
        $db = DB_NAME;
        $user = DB_USER;
        $pass = DB_PASS;
        $dsn = "mysql:host=$host;dbname=$db;charset=utf8";
        
        try{
            $dbh = new PDO($dsn, $user, $pass, [
                PDO::ATTR_ERRMODE => PDO::ERRMODE_EXCEPTION,
                PDO::ATTR_DEFAULT_FETCH_MODE =>PDO::FETCH_ASSOC
            ]);
        }catch(PDOException $e){
            echo '連接失敗' . $e->getMessage();
        }

        return $dbh;
    }

    // 取得所有文章列表
    function getBlogAll() {
        $dbh = connect();
        
        $sql = "SELECT * FROM blogs";
        $stmt = $dbh->query($sql);
        $result = $stmt->fetchAll(PDO::FETCH_ASSOC);

        return $result;
    }

    // 取得單個文章的詳細內容
    function getBlog() {
        $dbh = connect();

        $id = $_GET['id'];
        $sql = 'SELECT * FROM blogs WHERE id = :id';
        $stmt = $dbh->prepare($sql);
        $stmt->bindValue(':id', $id, PDO::PARAM_INT);
        $stmt->execute();
        $result = $stmt->fetch(PDO::FETCH_ASSOC);

        return $result;
    }

    // 判斷分類顯示
    function getCategory($category) {
        switch($category) {
            case 1:
                return '日常';
                break;
            case 2:
                return '音樂';
                break;
        }
    }

    // 判斷公開狀態
    function getPublishStatus($publish_statusy) {
        switch($publish_statusy) {
            case 1:
                return '公開';
                break;
            case 2:
                return '非公開';
                break;
        }
    }

    // 文章發表驗證
    function blogValidate() {
        if(empty($_POST['title'])) {
            exit('請輸入標題');
        }
        if(mb_strlen($_POST['title']) > 191) {
            exit('標題必須在191字以內');
        }
        if(empty($_POST['content'])) {
            exit('請輸入內文');
        }
    }

    // 新增文章
    function create_blog() {    
        $dbh = connect();
        $sql = 'INSERT INTO blogs(title, content, category, publish_status) VALUE (:title, :content, :category, :publish_status)';
    
        $dbh->beginTransaction();
        try{
            $stmt = $dbh->prepare($sql);
            $stmt->bindValue('title', $_POST['title'], PDO::PARAM_STR);
            $stmt->bindValue('content', $_POST['content'], PDO::PARAM_STR);
            $stmt->bindValue('category', $_POST['category'], PDO::PARAM_INT);
            $stmt->bindValue('publish_status', $_POST['publish_status'], PDO::PARAM_INT);
            $stmt->execute();
            $dbh->commit();
        }catch(PDOException $e) {
            $dbh->rollBack();
            exit('新增失敗');
        }
    
        echo '新增成功<br>';
    }

    // 編輯文章
    function update_blog() {    
        $dbh = connect();
        $sql = 'UPDATE blogs SET title = :title, content = :content, category = :category, publish_status = :publish_status WHERE id = :id';
    
        $dbh->beginTransaction();
        try{
            $stmt = $dbh->prepare($sql);
            $stmt->bindValue('id', $_POST['id'], PDO::PARAM_INT);
            $stmt->bindValue('title', $_POST['title'], PDO::PARAM_STR);
            $stmt->bindValue('content', $_POST['content'], PDO::PARAM_STR);
            $stmt->bindValue('category', $_POST['category'], PDO::PARAM_INT);
            $stmt->bindValue('publish_status', $_POST['publish_status'], PDO::PARAM_INT);
            $stmt->execute();
            $dbh->commit();
        }catch(PDOException $e) {
            $dbh->rollBack();
            exit('更新失敗' . $e);
        }
    
        echo '更新成功<br>';
    }

    // 刪除文章
    function delete() {
        $dbh = connect();
        $sql = 'DELETE FROM blogs WHERE id = :id';
        $stmt = $dbh->prepare($sql);
        $stmt->bindValue(':id', $_GET['id'], PDO::PARAM_INT);
        $stmt->execute();
        echo '刪除成功<br>';
    }
?>

# env.php

設定連線環境變數

    <?php
        define('DB_HOST', 'localhost');

        define('DB_USER', 'root');

    ?>

# index.php

    <?php
        require_once 'dbc.php';

        $blogData = getBlogAll();
    ?>

    <body>
        <a href="form.html">新增</a>
        <table>
            <tr>
                <th>標題</th>
                <th>分類</th>
                <th>狀態</th>
            </tr>
            <?php foreach($blogData as $value): ?>
            <tr>
                <td><?php echo $value['title'] ?></td>
                <td><?php echo getCategory($value['category']) ?></td>
                <td><?php echo getPublishStatus($value['publish_status']) ?></td>
                <td><a href="detail.php?id=<?php echo $value['id'] ?>">詳細</a></td>
                <td><a href="update_form.php?id=<?php echo $value['id'] ?>">編輯</a></td>
                <td><a href="delete.php?id=<?php echo $value['id'] ?>">刪除</a></td>
            </tr>
            <?php endforeach; ?>
        </table>
    </body>
