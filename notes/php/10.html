# dbc.php

<?php
    require_once 'env.php';
    
    function connect() {
        $host = DB_HOST;
        $db = DB_NAME;
        $user = DB_USER;
        $pass = DB_PASS;
        $dsn = "mysql:host=$host;dbname=$db;charset=utf8";
        
        try{
            $dbh = new PDO($dsn, $user, $pass, [
                PDO::ATTR_ERRMODE => PDO::ERRMODE_EXCEPTION,
                PDO::ATTR_DEFAULT_FETCH_MODE =>PDO::FETCH_ASSOC
            ]);
            
        }catch(PDOException $e){
            echo '連接失敗' . $e->getMessage();
        }

        return $dbh;
    }  
?>

    function h($str) { // 防止XXS攻擊
        return htmlspecialchars($str, ENT_QUOTES, 'utf-8');
    }
    
    function csrcToken() { // 設置token
        $csrf_toke = bin2hex(random_bytes(32));
        $_SESSION['csrf_token'] = $csrf_toke;
        return $csrf_toke;
    }
?>

    if(!$name = filter_input(INPUT_POST, 'name')) {

    if(!$email = filter_input(INPUT_POST, 'email')) {

# UserLogic.php

<?php
    require_once '../dbc.php';

    class UserLogic
    {
        public static function createUser($userData) // 註冊功能
        {
            $result = false;

            $arr = [];
            $arr[] = $userData['name'];
            $arr[] = $userData['email'];
            $arr[] = password_hash($userData['password'], PASSWORD_DEFAULT);

            $sql = 'INSERT INTO users(name, email, password) VALUES(?, ?, ?)';

            try{
                $stmt = connect()->prepare($sql);
                $result = $stmt->execute($arr);

                return $result;
            }catch(PDOException $e){
                return $result;
            }
        }

        public static function login($name, $password) // 登入功能
        {
            $result = false;

            $user = self::getUserByName($name);

            if(!$user) {
                $_SESSION['msg'] = '帳號不存在';
                return $result;
            }

            if(password_verify($password, $user['password'])) {
                session_regenerate_id(true); // 產生新的Session ID
                $_SESSION['user_login'] = $user;
                $result = true;
                return $result;
            }
            
            $_SESSION['msg'] = '密碼不正確';
            return $result;
        }

        public static function getUserByName($name) // 以使用者帳號來查詢使用者 
        {
            $result = false;

            $arr = [];
            $arr[] = $name;

            $sql = 'SELECT * FROM users WHERE name = ?';

            try{
                $stmt = connect()->prepare($sql);
                $stmt->execute($arr);
                $user = $stmt->fetch();
                return $user;
            }catch(PDOException $e){
                return $result;
            }
        }

        public static function checkLogin() { // 檢查是否為登入狀態
            $result = false;

            if(isset($_SESSION['user_login']) && $_SESSION['user_login']['id'] > 0) {
                return $result = true;
            }

            return $result;
        }
        
        public static function logout() { // 登出功能
            $_SESSION = array();
            session_destroy();
        }
    }
?>
