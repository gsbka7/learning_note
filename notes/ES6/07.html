# 3. this指向

不同的調用方式決定了 this 的指向不同

  1. 普通函式 指向 window
    function fn() {
    console.log(this);
};

  1. 物件 指向物件 o
var o = {
    say : function () {
        console.log(this);
    }
};
o.say();
  1. 構造函式 指向 star 這個實體物件, 物件原型裡的 this 指向的也是 star 這個實體物件

    }
};
var star = new Star();

new Star();
  1. 綁定事件函式 指向是函式的調用者 指向的是btn按鈕物件

  1. 定時器 指向window

# 4. 改變函式內this指向

call()

apply()

bind()

# 1. call()

可以調用函式，也可以改變函式內的 this指向

fun.call(thisArg, arg1, arg2, ...)

var call = {
    name: 'name'
};
function fn(a, b) {
    console.log(this);
    console.log(a + b);
};
fn.call(call, 1, 2); // this指向 o

// call的主要作用可以實現繼承
function Father(uname, age, sex) {
    this.uname = uname;
    this.age = age;
    this.sex = sex;
};
function Son(uname, age, sex) {
    // Father 的 this 修改為 Son 的 this
    // 子構造函式調用父構造函式的屬性
    Father.call(this, uname, age, sex) 
};
var son = new Son('father', '40', 'men');
console.log(son);

# 2. apply()

可以調用函式，也可以改變函式內的 this指向

fun.apply(thisArg, [argsArray])

・thisArg：fun函式運作時指定的 this 值

・argsArray：傳遞的值，必須使用陣列的方式

var apply = {
    name: 'name'
};
function fn(arr) {
    console.log(this);
    console.log(arr); // arr1

};
// 參數必須是陣列
fn.apply(apply, ['arr1']);

apply主要應用：

// Math.max() 求最大值
var arr = [1, 45, 99, 23, 3];
var max = Math.max.apply(Math, arr);
var min = Math.min.apply(Math, arr);
console.log(max, min);

# 3. bind()

不會調用原本的函式，可以改變原本函式內部的 this指向

返回的是原函式改變 this 之後產生的新函式

fun.bind(thisArg, arg1, arg2, ...)

・thisArg：fun函式運作時指定的 this 值

・arg1, arg2：傳遞的其他參數

var bind = {
    name: 'name'
};

function fn(a, b) {
    console.log(this);
    console.log(a + b);
};
var f = fn.bind(bind, 1, 2);
f();

應用：如果有的函式不需要立即調用, 但又想改變 this 指向

點擊後禁用按鈕, 3秒後重新啟用

// 錯誤
var btn = document.querySelector('button');
btn.onclick = function () {
    this.disabled = true; // this 指向 btn
    setTimeout(function () {
        this.disabled = false; // this 指向 window，不會有效果
    }.bind(this), 3000)
};

// 正確：宣告變數，修改 this 指向
var btn = document.querySelector('button');
btn.onclick = function () {
    this.disabled = true; // this 指向 btn
    var that = this; 
    setTimeout(function () {
        that.disabled = false; // this 指向 window
    }.bind(this), 3000)
};

// bind()
var btns = document.querySelectorAll('button');
for (var i = 0; i < btns.length; i++) {
    btns[i].onclick = function () {
        this.disabled = true;
        setTimeout(function () {
            this.disabled = false; // this 指向 window
        }.bind(this), 3000) // bind定義在定時器函式的外面，btn的裡面，所以會指向btn
    }
}

# 5. 總結

  1. 相同點

可以改變 this 指向

  1. 區別點

(1) call() appyl() 會調用函式, 並且改變 this 指向

(2) call() 和 apply() 傳遞參數不同, apply必須陣列形式

(3) bind() 不會調用函式

  1. 主要應用

(1) call() 繼承

(2) apply() 經常和陣列有關, 例如陣列中值的最大值與最小值

(3) bind() 不調用函式, 例如定時器中的 this 指向
