# 【 Promise 】

# 1. 介紹

解決非同步（asynchronous）以及callback hell(回呼地獄)的方案

# 1. 非同步解決方案

JavaScript在執行非同步操作時，先執行完的程式碼會先響應結果。

如果第二個工作的執行需要依賴第一個工作的結果，而第二個工作在第一個工作之前就先執行完畢回傳結果，會導致錯誤發生

使用Pormise可以確保每一個非同步操作都是依照順序執行的

# 2. callback hell(回呼地獄)

範例：多個串聯的非同步操作形成的回呼地獄

程式碼會不斷地向右擴展

    setTimeout(() => {
        console.log(1);
        setTimeout(() => {
            console.log(2);
            setTimeout(() => {
                console.log(3);
            }, 2000);
        }, 2000);
    }, 2000);

# 使用 Promise 解決方案

可讀性佳

var p = new Promise(function(resolve, reject) {
    setTimeout(function() {
        resolve('1')
    }, 2000)
})

// 實例化Promise
p.then(function(val){
    console.log(val)
    return new Promise(function(resolve, reject) { // 返回 Promise實例
        setTimeout(function() {
            resolve('2')
        }, 2000)
    })
}).then(function(val) { // 鏈式調用
    console.log(val)
})

# 2. 三種狀態

pending(進行中)、fulfilled(成功)、Rejected(失敗)

狀態不受外界影響，只有非同步操作的結果，決定目前是哪一種狀態

一旦狀態確定後就不會再改變

・ pending -> fulfilled

・ pending -> Rejected

# 3. 實例化

Promise是一個構造函式，用來建立Promise實例

const p = new Promise(function(resolve, reject) {
    // resolve：將Promise的狀態從未完成改變為成功，在非同步操作成功時調用
    resolve() // 返回非同步操作的結果，作為參數傳遞出去

    // reject：將Promise的狀態從未完成改變為失敗，在非同步操作失敗時調用
    reject() // 返回非同步操作的結果，作為參數傳遞出去

})

# 實例化方法

# 1. then()

當實例狀態發生改變時的回調函式，返回的是一個新的Promise實例，也就是Promise可以鏈式書寫的原因

const p = new Promise(function(resolve, reject) {
    setTimeout(() => {
        const time = new Date().getTime();
        if(time % 2 == 0) {
            resolve('成功的狀態' + time)
        }else{
            reject('失敗的狀態' + time)
        }
    }, 1000);
})

// 調用實例化物件
p.then((value) => { // resolve
    console.log('成功的資料' + value);
}),((reason) => { // reject
    console.log('失敗的資料' + reason);
}).then()

# 2. catch()

用於指定發生錯誤的回調函式

Promise實例的錯誤是有冒泡性質的，在鏈式調用中會不斷向後面傳遞，直到捕捉(catch)到錯誤為止

const p = new Promise(function(resolve, reject) {
    setTimeout(() => {
        const time = new Date().getTime();

        reject('失敗的狀態' + time)
    }, 1000);
})

// 調用實例化物件
p.catch((value) => {
    console.log(value);
})

# 3. finally()

不論Promise狀態是否失敗，都會執行操作

const p = new Promise(function(resolve, reject) {
    setTimeout(() => {
        const time = new Date().getTime();
        if(time % 2 == 0) {
            resolve('成功的狀態' + time)
        }else{
            reject('失敗的狀態' + time)
        }
    }, 1000);
})

// 調用實例化物件
p.then((value) => {
    console.log(value);
}).catch((value) => {
    console.log(value);
}).finally(() => {
    console.log('最後的結果');
})

# 4. 回調地域

範例：一秒後輸出1，二秒後輸出2，三秒後輸出3

可讀性不佳

    setTimeout(() => {
        console.log(1);
        // 在這裡處裡第一個非同步操作

        setTimeout(() => {
            console.log(2);
            // 在這裡處裡第二個非同步操作

            setTimeout(() => {
                console.log(3);
                // 在這裡處裡第三非同步操作

            }, 3000);
        }, 2000);
    }, 1000);

使用

    function getDate(){
        return new Promise((resolve, reject) => {
            setTimeout(() => {
                console.log(1);
                resolve(2) // 將參數傳遞出去
            }, 1000);
        })
    }

    // 鏈式調用
    getDate().then((value) => { // 接收resolve的參數
    console.log(val)
        // 返回 Promise實例
        return new Promise((resolve, reject) => { // 如果在這裡沒有使用 return 且使用resolve會出現 is not defined
            setTimeout(() => {
            console.log(value);
            resolve(3)
        }, 2000);
        })
    }).then((value) => {
        return new Promise((resolve, reject) => {
            setTimeout(() => {
            console.log(value);
        }, 3000);
        })
    })

# 5. Promise.all

用於將多個Promise實例封裝成一個新的Promise實例

let p1 = new Promise((resolve, reject) => {
    resolve('成功01')
})
let p2 = new Promise((resolve, reject) => {
    resolve('成功02')
})
let p3 = new Promise((resolve, reject) => {
    resolve('成功03')
})

// 參數可以不是陣列，但必須是interator
let pAll = Promise.all([p1, p2, p3])
console.log(pAll);
// pAll的狀態是由p1、p2、p3來決定的，當這三個的都會成功，pAll才會成功；有一個失敗，pAll就會失敗，會返回第一個失敗的實例的返回值

pAll.then((value) => {
        console.log(value);
    }).catch((reason) => {
        console.log(reason);
    })

如果作為參數的實例，自己定義了catch方法，一旦rejected，不會觸發pAll的catch方法

let p2 = new Promise((resolve, reject) => {
        // resolve('成功02')
        reject('失敗02')
    }).catch((reason) => {
        console.log(reason); 
    })

# 將多個請求結果合併在一起

function getUser() {
    return new Promise((resolve, reject) => {
        setTimeout(() => {
            resolve('會員資料')
    }, 1000);
    })
}
function getCategory() {
    return new Promise((resolve, reject) => {
        setTimeout(() => {
            resolve('歌曲分類')
        }, 2000);
    })
}
function getList() {
    return new Promise((resolve, reject) => {
        setTimeout(() => {
            resolve('歌曲列表')
    }, 3000);
    })
}

function initload() {
    let All = Promise.all([getUser(), getCategory(), getList()])
    console.log(All);

    All.then((value) => {
        console.log(value); // 當三個Promise都成功時才會串入參數渲染畫面
    })
}
initload();

# 6. Promise.race()

將多個Promise實例封裝成一個新的Promise實例

範例：

只要實例中有一個先改變狀態，就會把這個實例的返回值傳給prace的回調函式

let p1 = new Promise((resolve, reject) => {
    setTimeout(() => {
        resolve('p1成功')
    }, 2000);
})
let p2 = new Promise((resolve, reject) => {
    setTimeout(() => {
        resolve('p2成功')
    }, 1000);
})
const prace = Promise.race([p1, p2]);
console.log(prace); // p2成功

提示

Promise.all會返回陣列，Promise.race只會返回一個結果

範例：請求超時提示

function request() {
    return new Promise((resolve, reject) => {
        setTimeout(() => {
            resolve('請求成功')
        }, 4000);
    })
}
function timeout() {
    return new Promise((resolve, reject) => {
        setTimeout(() => {
            reject('請求失敗')
        }, 3000);
    })
}

Promise.race(request(), timeout())
.then((value) => {
    console.log(value);
}).catch((reason) => {
    console.log(reason);
})

# 7. Promise封裝Ajax

function getJson(url) {
    return new Promise((resolve, reject) => {
        let xhr = new XMLHttpRequest();
        xhr.open('GET', url, true); // 第三個參數：是否執行非同步操作
        // 發送http請求
        xhr.send(null);
        // 設定狀態的監聽函式
        xhr.onreadystatechange = function() {
            if(xhr.readyState !== 4) return; // 表示請求成功

            // 當請求成功或者失敗，需要改變Promise狀態
            if(xhr.status >= 200 && xhr.status < 300) {
                resolve(xhr.response)
                // resolve(xhr.responseText) // 如果沒有設置資料的類型
            }else{
                reject(new Error(xhr.statusText))
            }
        }

        // 設置錯誤的監聽函式
        xhr.onerror = function() {
            reject(new Error(xhr.statusText))
        }

        // 設置響應資料的類型
        xhr.responseType = 'json'
    })
}

getJson('https://pokeapi.co/api/v2/pokemon/ditto').then((value) => {
        console.log(value);
    }).catch((reason) => {
        console.log(reason);
    })
