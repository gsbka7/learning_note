# 01 【 類 】

# 1. 建立類和實體化物件

語法：

class Name {
    
};

實體化：類必須使用 new 實體化物件

var name = new Name();

# 1. 建立類 class

・ 通過 class 建立類，首字母大寫

・ 類裡面有個 constructor 函式 可以接受傳遞過來的參數, 同時返回物件實例

・ 只要 new 產生實體化時, 就會自動調用這個函式，即使不寫上也會自動產生

・ 類裡面的所有函式不需要寫function

・ 多個方法和函式不需要加逗號分割

class Star {
    constructor(uname, age) {
        this.uname = uname
        this.age = age;
    };
    sing(song) {
        console.log(this.uname + song);
    };
};

# 2. 使用 類 建立 物件

・ new 不能省略

var star1 = new Star('star1', '18');
var star2 = new Star('star2', '20');
star1.sing('song1');
star2.sing('song2');

# 02 繼承

# 1. 類的繼承

  1. extends

子類繼承父類的方法

class Father {
    constructor() {

    };
    
    money() {
        console.log(100);
    };
};
class Son extends Father { // Son 繼承 Father 的屬性和方法

};

var son = new Son();
son.money(); // 使用 Father 的 money()
  1. super

子類向父類傳遞參數

class Father {
    constructor(x, y) {
        this.x = x;
        this.y = y;
    };

    sum() {
        console.log(this.x + this.y);
    };
};

class Son extends Father {
    constructor(x, y) {
        // 參數無法傳遞給父類
        // this.x = x;
        // this.y = y;

        super(x, y) // 調用父類中的構造函式
    };
};

var son = new Son(1, 2);
son.sum();

# 03 super調用父類中的函式

class Father {
    say() {
        return 'Father';
    };
};

class Son extends Father {
    say() {
        console.log('son'); // son，就近原則
        console.log(super.say()); // 調用父類中的函式
    };
};

var son = new Son();
son.say();

// 就近原則, 先看子類有無此方法, 有則執行, 沒有則從父類找

# 04 子類繼承父類同時擴展自己

// 父類有 sum 方法
class Father {
    constructor(x, y) {
        this.x = x;
        this.y = y;
    };

    sum() {
        console.log(this.x + this.y);
    }
};

// 子類繼承 sum 方法 同時擴展 sub 方法
class Son extends Father {
    constructor(x, y) {
        // 利用 super 調用父類的構造函式
        // 必須在 子類this 前調用
        super(x.y);
        
        this.x = x;
        this.y = y;
    };

    sub() {
        console.log(this.x - this.y);
    };
};

var son = new Son(5,3);
son.sub(); // 2
son.sum(); // 8

# 05 使用類的注意事項

<body>
    <button>點擊</button>
</body>

var that;
class Star {
    constructor(uname, age) {
        // constructor 裡的 this 指向的是 物件實例
        that = this;
        console.log(this);

        this.uname = uname;
        this.age = age;
        // this.sing(); // 調用 sing 需要加 this

        this.btn = document.querySelector('button');
        this.btn.onclick = this.sing; // 加上()會立即調用，不加會等點擊後才調用
    };

    sing() {
        // sing方法 的 this 會指向 btn
        // that 儲存的是 construtor 裡的 this
        console.log(that.uname);
    }
};

var star1 = new Star('star1'); // 位置必須在類的後面
console.log(that === star1); // true

// 1.在 ES6 中沒有變數提升, 所以必須先定義類, 才能實體化物件
// 2.類裡面共有的屬性和方法一定要加 this 使用

# 06 類的本質

類的本質其實還是函式, 也可以認為是 構造函式 的另一種寫法

ES6 之前通過 構造函式 + 原型 實現 面向物件程式設計

// ES6 通過 類 實現 面向物件程式設計
class Star {

};

console.log(typeof Star); // function

# 1.1 構造函式的特點

・・ ES6 之前通過 構造函式 + 原型 實現 面向物件程式設計

・・ 1. 有物件原型 prototype

・・ 2. 物件原型prototype 裡面有 constructor 指向構造函式本身

・・ 3. 可以通過 物件原型 新增方法

・・ 4. 建立的實體物件有 proto 原型指向 構造函式的物件原型

# 1.2 類

ES6 的 類 絕大部分功能, ES5 都可以做到, 新的 class 寫法只是讓物件原型更加清晰, 更像 面向物件程式設計

可以視為語法糖

・・ 1.構造函式 有物件原型 prototype

console.log(Star.prototype);

・・ 2. 物件原型prototype 裡面有 constructor 指向構造函式本身

console.log(Star.prototype.constructor);

・・ 3. 可以通過 物件原型 新增方法

Star.prototype.sing = function () {
    console.log('sing');
};

var star1 = new Star();
console.log(star1);

・・ 4. 建立的實體物件有 proto 原型指向 構造函式的物件原型

console.log(star1.__proto__ === Star.prototype);
