# 04 【 BeanFactory、ApplicationContext 】

# 1. BeanFactory入門

# 1. 開發步驟

  1. 導入Spring的jar包或Maven座標

  2. 定義UserService接口及UserServiceImpl實現類

  3. 建立bean.xml配置文件，配置UserServiceImpl

  4. 編寫測試，建立BeanFactory，加載配置文件，取得UserService物件

    <dependencies>
        <dependency>
            <groupId>org.springframework</groupId>
            <artifactId>spring-context</artifactId>
            <version>5.3.26</version>
        </dependency>
    </dependencies>

package com.sp.service;

public interface UserService {}

package com.sp.service.impl;

import com.sp.service.UserService;

public class UserServiceImpl implements UserService {}

<bean id="userService" class="com.sp.service.impl.UserServiceImpl"></bean>

package com.sp.service.test;

import com.sp.service.UserService;
import org.springframework.beans.factory.support.DefaultListableBeanFactory;
import org.springframework.beans.factory.xml.XmlBeanDefinitionReader;

public class Test {
    public static void main(String[] args) {
        // 1. 建立工廠物件
        DefaultListableBeanFactory beanFactory = new DefaultListableBeanFactory();
        // 2. 建立一個讀取器(讀取xml)
        XmlBeanDefinitionReader reader = new XmlBeanDefinitionReader(beanFactory);
        // 3. 讀取配置文件給工廠
        reader.loadBeanDefinitions("beans.xml");
        // 4. 根據id取得bean實體物件
        UserService userService = (UserService) beanFactory.getBean("userService");
        System.out.println(userService);
    }
}

# 2. set方法注入

    <dependencies>
        <dependency>
            <groupId>org.springframework</groupId>
            <artifactId>spring-context</artifactId>
            <version>5.3.26</version>
        </dependency>
    </dependencies>

package com.sp.service;

public interface UserService {}

package com.sp.service.impl;

import com.sp.dao.UserDao;
import com.sp.service.UserService;

public class UserServiceImpl implements UserService {

    private UserDao userDao;

    // set方法，UserService依賴UserDao的注入
    // BeanFactory去調用該方法，從容器(BeanFactory)中取得UserDao設置到此處
    public void setUserDao(UserDao userDao) {
        System.out.println("BeanFactory去調用該方法取得UserDao設置到此處"+userDao);

        // 賦值給成員變數
        this.userDao = userDao;
    }
}

<bean id="userService" class="com.sp.service.impl.UserServiceImpl">
    <!--name：set方法名，去掉set，首字母小寫-->
    <!--ref：要注入的bean的id-->
    <property name="userDao" ref="userDao"></property>
</bean>

<bean id="userDao" class="com.sp.dao.impl.UserDaoImpl"></bean>

# 2. ApplicationContext入門

# 1. 使用

ApplicationContext稱為Spring容器，內部封裝了BeanFactory

使用ApplicationContext進行開發時，xml配置文件名稱習慣寫成ApplicationContext.xml

    <dependencies>
        <dependency>
            <groupId>org.springframework</groupId>
            <artifactId>spring-context</artifactId>
            <version>5.3.26</version>
        </dependency>
    </dependencies>

package com.sp.service;

public interface UserService {}

package com.sp.service.impl;

import com.sp.dao.UserDao;
import com.sp.service.UserService;

public class UserServiceImpl implements UserService {

    private UserDao userDao;

    // set方法，UserService依賴UserDao的注入
    // BeanFactory去調用該方法，從容器(BeanFactory)中取得UserDao設置到此處
    public void setUserDao(UserDao userDao) {
        System.out.println("BeanFactory去調用該方法取得UserDao設置到此處"+userDao);

        // 賦值給成員變數
        this.userDao = userDao;
    }
}

<bean id="userService" class="com.sp.service.impl.UserServiceImpl">
    <!--name：set方法名，去掉set，首字母小寫-->
    <!--ref：要注入的bean的id-->
    <property name="userDao" ref="userDao"></property>
</bean>

<bean id="userDao" class="com.sp.dao.impl.UserDaoImpl"></bean>

package com.sp.test;

import com.sp.service.UserService;
import org.springframework.context.ApplicationContext;
import org.springframework.context.support.ClassPathXmlApplicationContext;

public class ApplicationContextTest {
    public static void main(String[] args) {
        
        // 1.取得Spring容器
        // ApplicationContext就是Spring容器，是一個接口
        // 接口中有許多實現類，其中一個實現類ClassPathXmlApplicationContext
        // ClassPathXmlApplicationContext用於加載spring配置文件(spring.xml)
        // 只要執行，就相當於啟動Spring容器，解析spring.xml，並且實體化所有bean物件，放到spring容器中
        ApplicationContext applicationContext = new ClassPathXmlApplicationContext("ApplicationContext.xml");

        // 2.根據bean的id從Spring容器中取得這個物件 
        UserService userService = (UserService) applicationContext.getBean("userService");

        // 3.輸出
        System.out.println(userService+"ApplicationContext");
    }
}

# 3. BeanFactory 與 ApplicationContext 的關係

  1. BeanFactory是Spring的早期接口，稱為BeanFactory的Bean工廠。 ApplicationContext是後期更高級的接口，稱為Spring容器

  2. ApplicationContext在BeanFactory基礎上對功能進行了擴展。BeanFactory的API偏向底層，ApplicationContext的API大多數是對這些底層API進行封裝

  3. Bean建立的主要邏輯和功能都被封裝在BeanFactory中，ApplicationContext不僅繼承了BeanFactory，而且內部還維護著BeanFactory的引用，所以兩者既有繼承關係，也有融合關係

  4. Bean的初始化時機不同，原始BeanFactory是在首次調用getBean時才進行Bean的建立；而ApplicationContext則是配置文件加載，容器建立就將Bean實體化並初始化好

// ApplicationContext在執行下面程式碼時就會實體化物件
new ClassPathXmlApplicationContext("ApplicationContext.xml");
