# 07 【 AOP 】

# 1. 介紹

# 1. AOP 概念

AOP，Aspect-Oriented Programming，面向導向程式設計

AOP是對OOP(物件導向程式設計)的補充延伸。

AOP底層使用動態代理進行實現

在不改變原始設計的基礎上為其進行功能增強

將與主要邏輯無關的程式碼抽取出來，形成一個獨立的組件，然後以橫向交叉的方式應用到流程當中，此過程稱為AOP

# 2. AOP 思想的實現

動態代理技術，在執行過程中，對目標物件的方法進行增強，代理物件同名方法內可以執行原有邏輯的同時，嵌入執行其他增強邏輯或其他物件的方法

對A物件產生Proxy物件

Proxy物件內部的方法與A物件內部的方法命名是相同的，類型也是相同的

例如A物件為UserService，Proxy物件也是UserService

            <!--如果只有一個通知-->
            <aop:before method="beforeAdvice" pointcut="execution(void com.spring.service.impl.UserServiceImpl.show2())"></aop:before>

# 2. 詳解

execution([訪問修飾符] 返回值類型 package.類名.方法名(參數))

execution(void com.spring.service.UserServiceImpl.show2())

// 表示UserServiceImpl類中的任意方法
execution(* com.spring.service.UserServiceImpl.*(..))

// 表示service下任意類的任意方法
execution(* com.spring.service.*.*(..))

// 表示任意package中的任意類的任意方法
execution(* *..*.*(..))

・ 訪問修飾符可以省略

・ *表示任意

・ package與類名之間使用單點，表示該package下的類。使用雙點..，表示該package及其子包下的類

・ 參數列表可以使用雙點..，表示任意參數

package com.spring.advice;

import org.aspectj.lang.ProceedingJoinPoint;

// 增強方法
public class MyAdvice {
    public void beforeAdvice() {
        System.out.println("前置增強方法");
    }

    public void afterReturningAdvice() {
        System.out.println("後置增強方法");
    }

    public Object around(ProceedingJoinPoint proceedingJoinPoint) throws Throwable {
        System.out.println("環繞前的方法");
        Object res = proceedingJoinPoint.proceed(); // 執行目標方法
        System.out.println("環繞後的方法");
        return res;
    }

    public void afterThrowingAdvice() {
        System.out.println("異常拋出通知");
    }

    public void afterAdvice() {
        System.out.println("最終通知");
    }
}

public class MyAdvice implements MethodBeforeAdvice , AfterReturningAdvice {

# 5. 兩種語法的不同點

# 1. 語法形式不同

・ advisor是通過實現接口來確認通知的類型

・ aspect是通過配置確認通知的類型，更加靈活

# 2. 可配置的切面數量不同：

・ 一個advisor只能配置一個固定通知和一個切點表達式

・ 一個aspect可以配置多個通知和多個切點表達式任意組合

# 3. 使用場景不同：

・ 允許隨意搭配情況下可以使用aspect進行配置

・ 如果通知類型單一、切面單一的情況下可以使用advisor進行配置

・ 在通知類型已經固定，不用人為指定通知類型時，可以使用advisor進行配置

# 2. 進階

package com.spring.advice;

import org.aspectj.lang.ProceedingJoinPoint;
import org.aspectj.lang.annotation.*;
import org.springframework.stereotype.Component;

// 增強方法
@Component
@Aspect
public class MyAdvice {
    // 表達式抽取
    @Pointcut("execution(* com.spring.service.impl.UserServiceImpl.*(..))")
    public void myPointcut() {}

    @Before("execution(* com.spring.service.impl.UserServiceImpl.*(..))")
    public void beforeAdvice() {
        System.out.println("前置增強方法");
    }

    @AfterReturning("execution(* com.spring.service.impl.UserServiceImpl.*(..))")
    public void afterReturningAdvice() {
        System.out.println("後置增強方法");
    }

    // 表達式抽取
    @Around("MyAdvice.myPointcut()")
    public Object around(ProceedingJoinPoint proceedingJoinPoint) throws Throwable {
        System.out.println("環繞前的方法");
        Object res = proceedingJoinPoint.proceed(); // 執行目標方法
        System.out.println("環繞後的方法");
        return res;
    }

    @AfterThrowing(pointcut = "execution(* com.spring.service.impl.UserServiceImpl.*(..))", throwing = "e")
    public void afterThrowingAdvice(Throwable e) {
        System.out.println(e);
        System.out.println("異常拋出通知");
    }

    @After("execution(* com.spring.service.impl.UserServiceImpl.*(..))")
    public void afterAdvice() {
        System.out.println("最終通知");
    }
}

# isolation屬性

指定交易管理的隔離級別，交易管理並行存在三大問題：髒讀(dirty read)、幻影讀(phantom read)、不可重複讀(non-repeatable reads)。可以通過設置隔離級別來保證並行問題的出現，常用的是READ_COMMITTED和REPEATABLE_READ

髒讀(dirty read)：當一個事務允許讀取另外一個事務修改但未提交的資料時，就可能發生髒讀

幻影讀(phantom read)：在事務執行過程中，當兩個完全相同的查詢語句執行得到不同的結果集

不可重複讀(non-repeatable reads)：在一次事務中，當一行資料取得兩遍得到不同的結果

package com.spring.config;

import com.alibaba.druid.pool.DruidDataSource;
import org.mybatis.spring.SqlSessionFactoryBean;
import org.mybatis.spring.annotation.MapperScan;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.ComponentScan;
import org.springframework.context.annotation.Configuration;
import org.springframework.context.annotation.PropertySource;
import org.springframework.jdbc.datasource.DataSourceTransactionManager;
import org.springframework.transaction.annotation.EnableTransactionManagement;

import javax.sql.DataSource;

@Configuration // 配置類
@ComponentScan("com.spring") // 組件掃描
@PropertySource("classpath:jdbc.properties") // 載入jdbc.properties
@MapperScan("com.spring.mapper") // mapper掃描
@EnableTransactionManagement // <tx:annotation-driven transaction-manager="transactionManager"/>
public class SpringConfig {

    @Bean
    public DataSource dataSource(
            @Value("${jdbc.driver}") String driver,
            @Value("${jdbc.url}") String url,
            @Value("${jdbc.username}") String username,
            @Value("${jdbc.password}") String password
    ) {
        DruidDataSource dataSource = new DruidDataSource();
        dataSource.setDriverClassName(driver);
        dataSource.setUrl(url);
        dataSource.setUsername(username);
        dataSource.setPassword(password);
        return dataSource;
    }

    @Bean
    public SqlSessionFactoryBean sqlSessionFactoryBean(DataSource dataSource) {
        SqlSessionFactoryBean sqlSessionFactoryBean = new SqlSessionFactoryBean();
        sqlSessionFactoryBean.setDataSource(dataSource);
        return sqlSessionFactoryBean;
    }

    @Bean
    public DataSourceTransactionManager transactionManager(DataSource dataSource) {
        DataSourceTransactionManager dataSourceTransactionManager = new DataSourceTransactionManager();
        dataSourceTransactionManager.setDataSource((dataSource));
        return dataSourceTransactionManager;
    }
}
